                    </div> <h1 style="font-family: Arial, Helvetica, Sans-Serif; font-size: 18px; font-weight: bold; color: #FF6600; margin-left:10px;margin-top:10px;" class="">QuickSort - Sort Arrays using the QuickSort Algorithm</h1> 
                    <div style="padding:10px;"> 
                     <font color="DarkBlue"><pre><code><font color="Green">' QuickSort an array of any type</font>
<font color="Green">' QuickSort is especially convenient with large arrays (&gt;1,000</font>
<font color="Green">' items) that contains items in random order. Its performance</font>
<font color="Green">' quickly degrades if the array is already almost sorted. (There are</font>
<font color="Green">' variations of the QuickSort algorithm that work good with </font>
<font color="Green">' nearly-sorted arrays, though, but this routine doesn't use them.)</font>
<font color="Green">'</font>
<font color="Green">' NUMELS is the index of the last item to be sorted, and is</font>
<font color="Green">' useful if the array is only partially filled.</font>
<font color="Green">'</font>
<font color="Green">' Works with any kind of array, except UDTs and fixed-length</font>
<font color="Green">' strings, and including objects if your are sorting on their</font>
<font color="Green">' default property. String are sorted in case-sensitive mode.</font>
<font color="Green">'</font>
<font color="Green">' You can write faster procedures if you modify the first two lines</font>
<font color="Green">' to account for a specific data type, eg.</font>
<font color="Green">' Sub QuickSortS(arr() As Single, Optional numEls As Variant,</font>
<font color="Green">' ' Optional descending As Boolean)</font>
<font color="Green">' Dim value As Single, temp As Single</font>

Sub QuickSort(arr As Variant, Optional numEls As Variant, _
    Optional descending As Boolean)

    Dim value As Variant, temp As Variant
    Dim sp As Integer
    Dim leftStk(32) As Long, rightStk(32) As Long
    Dim leftNdx As Long, rightNdx As Long
    Dim i As Long, j As Long

    <font color="Green">' account for optional arguments</font>
    If IsMissing(numEls) Then numEls = UBound(arr)
    <font color="Green">' init pointers</font>
    leftNdx = LBound(arr)
    rightNdx = numEls
    <font color="Green">' init stack</font>
    sp = 1
    leftStk(sp) = leftNdx
    rightStk(sp) = rightNdx

    Do
        If rightNdx &gt; leftNdx Then
            value = arr(rightNdx)
            i = leftNdx - 1
            j = rightNdx
            <font color="Green">' find the pivot item</font>
            If descending Then
                Do
                    Do: i = i + 1: Loop Until arr(i) &lt;= value
                    Do: j = j - 1: Loop Until j = leftNdx Or arr(j) &gt;= value
                    temp = arr(i)
                    arr(i) = arr(j)
                    arr(j) = temp
                Loop Until j &lt;= i
            Else
                Do
                    Do: i = i + 1: Loop Until arr(i) &gt;= value
                    Do: j = j - 1: Loop Until j = leftNdx Or arr(j) &lt;= value
                    temp = arr(i)
                    arr(i) = arr(j)
                    arr(j) = temp
                Loop Until j &lt;= i
            End If
            <font color="Green">' swap found items</font>
            temp = arr(j)
            arr(j) = arr(i)
            arr(i) = arr(rightNdx)
            arr(rightNdx) = temp
            <font color="Green">' push on the stack the pair of pointers that differ most</font>
            sp = sp + 1
            If (i - leftNdx) &gt; (rightNdx - i) Then
                leftStk(sp) = leftNdx
                rightStk(sp) = i - 1
                leftNdx = i + 1
            Else
                leftStk(sp) = i + 1
                rightStk(sp) = rightNdx
                rightNdx = i - 1
            End If
        Else
            <font color="Green">' pop a new pair of pointers off the stacks</font>
            leftNdx = leftStk(sp)
            rightNdx = rightStk(sp)
            sp = sp - 1
            If sp = 0 Then Exit Do
        End If
    Loop
End Sub
