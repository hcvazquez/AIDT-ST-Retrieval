&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; return true;

                   <br>The functions behind that are:
                   <br>
                   <!--[CodeBlockStart:fc842fee-9b3e-4441-a961-a727055ffc84][excluded]-->
                   <pre class="plain" name="code">

function isEmailValid(value){
var problemMsg = isEmailValidWithReason(value);
if (problemMsg != "") {
jslog("isEmailValid: " + problemMsg);
return false;
}
return true;
}

// validate email address and return a reason why it is not valid
function isEmailValidWithReason(value) {
var localPartChars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!#$%*/?|^{}`~&amp;'+-=_.";
var domainChars =&nbsp;&nbsp;&nbsp; "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-_.";

// break into local part and domain
if (value.indexOf("@") == -1)
return "missing @ sign";
var s = value.split("@");
if (s.length != 2)
return "too many at signs";

// check the local part of the address
if (!containsOnlyChars(localPartChars, s[0]))
return "invalid character before the at sign";
if (s[0].length &lt; 1)
return "at least one character must be before the at sign";
if (s[0].substr(0,1) == ".")
return "period cannot be the first character";
if (s[0].substr(s[0].length-1,1) == ".")
return "period cannot be the last character before the at sign";

// check the domain part of the address
if (!containsOnlyChars(domainChars, s[1]))
return "invalid character after the at sign";
var periodIndex = s[1].indexOf(".");
if (periodIndex == -1)
return "missing period after the at sign";
if (periodIndex == 0)
 return "period cannot be the first character after the at sign";
var periods = s[1].split(".");
var lastPeriod = periods[periods.length-1];
if (lastPeriod.length &lt; 2)
return "must be at least 2 characters after the last period";
if (!isAlphaNum(s[1].substr(0,1)))
return "the first character after the at sign must be alphanumeric";
if (!isAlphaNum(s[1].substr(s[1].length-1,1)))
return "the last character must be alphanumeric";
return ""; // address is OK
}
</pre>

                 <section> 
                  <!-- [DocumentBodyStart:0d6d2349-c989-49d0-b10a-be51cad6874a] -->
                  <div class="jive-rendered-content">
                   <p>I use the below as an onchange script for the field being validated.. one thing you may want to incorporate that i did was the var err_field=control.id;</p>
                   <p style="min-height: 8pt; padding: 0px;">&nbsp;</p>
                   <p>this makes the code portable.. you just drop it into the onchange script and set the field you are watching and it will get the name of the field for the script by itself.</p>
                   <p style="min-height: 8pt; padding: 0px;">&nbsp;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">function onChange(control, oldValue, newValue, isLoading) {</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp; if (isLoading || newValue == '') {</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; return;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp; }</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; //validates a valid email address for the onchange field</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; var err_field = control.id;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; var err_message = 'Invalid Email Address (</span><a class="jive-link-email-small" href="mailto:abc@domain.com">abc@domain.com</a><span>)',</span></p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; regex = /[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,4}/ig;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; var err_flag = 'false';</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; err_flag = regex.test(g_form.getValue(err_field));</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; g_form.hideErrorBox(err_field);</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; if (!err_flag) {</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; g_form.showFieldMsg(err_field, err_message, 'error');</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; }</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; else { g_form.hideErrorBox(err_field); }</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">}</p>
                   <p style="min-height: 8pt; padding: 0px; font-family: Calibri; font-size: 11.0pt;">&nbsp;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">for those that want it the below does phone numbers &lt;us only&gt;</p>
                   <p style="min-height: 8pt; padding: 0px; font-family: Calibri; font-size: 11.0pt;">&nbsp;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">On change script</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">function onChange(control, oldValue, newValue, isLoading) {</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp; if (isLoading || newValue == '') {</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; return;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp; }</p>
                   <p style="min-height: 8pt; padding: 0px;">&nbsp;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp; //checks the variable the onchange is for to verify it is a valid phone number</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; var phone_field = control.id;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; g_form.hideFieldMsg(phone_field);</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; var intNum = g_form.getValue(phone_field);</p>
                   <p style="min-height: 8pt; padding: 0px;">&nbsp;</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; if (!isPhoneDash(intNum))</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; {</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;"> g_form.showFieldMsg(phone_field,'Invalid Phone Number Value ###-###-####','error');</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; }</p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">&nbsp;&nbsp; </p>
                   <p style="font-family: Calibri; font-size: 11.0pt;">}</p>

                      <div class="jive-rendered-content">
                       <p>I use this in the Service Portal, and it works perfectly.</p>
                       <p style="min-height: 8pt; padding: 0px;">&nbsp;</p>
                       <p>function onChange(control, oldValue, newValue, isLoading) {</p>
                       <p>&nbsp;&nbsp; if (isLoading || newValue == '') {</p>
                       <p>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; return;</p>
                       <p>&nbsp;&nbsp; }</p>
                       <p style="min-height: 8pt; padding: 0px;">&nbsp;</p>
                       <p>&nbsp; var email = newValue + '';</p>
                       <p>&nbsp; var regEx = new RegExp(/^(([^&lt;&gt;()\[\]\\.,;:\s@"]+(\.[^&lt;&gt;()\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/);</p>
                       <p>&nbsp; g_form.hideFieldMsg('u_personal_email');</p>
                       <p style="min-height: 8pt; padding: 0px;">&nbsp;</p>
                       <p>&nbsp; if (!regEx.test(email)) {</p>
                       <p>&nbsp; g_form.setValue('u_personal_email', '');</p>
                       <p>&nbsp; g_form.showFieldMsg('u_personal_email', 'Please enter a valid email address', 'error');</p>
                       <p>&nbsp; }</p>
                       <p>}</p>
