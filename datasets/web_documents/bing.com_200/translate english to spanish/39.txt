		  function check_if_url_starts_with_language_code()
		  {
			  var starts_with_language_code = false;
			  var language_code_found = '';

			  var pathname_parts = location.pathname.split('/');
			  
			  for(var i = 0; i < wt_languages.length; i++)
			  {
				  // check if it starts or ends with the language segment
				  if(pathname_parts[1] == wt_languages[i].key || pathname_parts[pathname_parts.length - 1] == wt_languages[i].key)
				  {
					  starts_with_language_code = true;
					  language_code_found = wt_languages[i].key;
				  } 
			  }

			  return {
				  starts_with_language_code: starts_with_language_code,
				  language_code_found: language_code_found
			  };

		  function update_url(language_code)
