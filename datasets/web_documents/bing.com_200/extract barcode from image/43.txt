function cookiesEnabled() {

function setDomainLanguage(language) {
if($.cookie("lang") != null) {

// do not display the domain selector if querystring is not present

// or if cookies are disabled
var displayDomainSelector = true;
function detectDomain() {
var queries = {};
$.each(document.location.search.substr(1).split('&'), function(c,q){
if(q != undefined && q != "") {
var i = q.split('=');
if(i[1] != undefined) {
queries[i[0].toString()] = i[1].toString();
}
}
});
var requested_domain = (queries['requestedDomain'] != undefined) ? queries['requestedDomain'] : "";
var nocookie = (queries['nocookie'] != undefined) ? queries['nocookie'] : "";
var client = (queries['client'] != undefined) ? queries['client'] : "";
if(requested_domain == "" || nocookie == "true" || !cookiesEnabled() || client == "addons" || window.overrideDomainSelector) {
displayDomainSelector = false;
}
}
detectDomain();
function extractDomain(url) {
var domain;
//find & remove protocol (http, ftp, etc.) and get domain
if (url.indexOf("://") > -1) {
domain = url.split('/')[2];
} else {
domain = url.split('/')[0];
}
//find & remove port number
domain = domain.split(':')[0];
return domain;
}

             <div id="body_question_110412" class="panel panel-default add_text_wrapping"> 
              <div class="panel-body"> 
               <img src="/matlabcentral/answers/uploaded_files/6070/image_a.JPG">
               <p>this code extract object random i want to extract letter line by line sequential</p>
               <pre class="language-matlab">%% Image segmentation and extraction
%% Read Image
imagen=imread('image_a.jpg');
%% Show image
figure(1)
imshow(imagen);
title('INPUT IMAGE WITH NOISE')
%% Convert to gray scale
if size(imagen,3)==3 % RGB image
  imagen=rgb2gray(imagen);
end
%% Convert to binary image
threshold = graythresh(imagen);
imagen =~im2bw(imagen,threshold);
%% Remove all object containing fewer than 30 pixels
imagen = bwareaopen(imagen,30);
pause(1)
%% Show image binary image
figure(2)
imshow(~imagen);
title('INPUT IMAGE WITHOUT NOISE')
%% Label connected components
[L Ne]=bwlabel(imagen);
%% Measure properties of image regions
propied=regionprops(L,'BoundingBox');
hold on
%% Plot Bounding Box
for n=1:size(propied,1)
  rectangle('Position',propied(n).BoundingBox,'EdgeColor','g','LineWidth',2)
end
hold off
pause (1)
%% Objects extraction
figure
for n=1:Ne
  [r,c] = find(L==n);
  n1=imagen(min(r):max(r),min(c):max(c));
  imshow(~n1);
  pause(0.5)
end
</pre>

              <div class="panel-body"> 
               <p>It's not random, though you don't know the algorithm it uses (it's top left to bottom right, column by column).</p>
               <p>You will need to find each line and extract it before you label. See attached demo, which will produce this:</p>
               <img src="/matlabcentral/answers/uploaded_files/6072/test.png">
               <p>There is not OCR package built in to MATLAB. You'll have to find one or do it yourself.</p>
