              <h4>Short Description:</h4> Natural Language Processing, Text Processing and Sentiment Analysis with Data at Rest and Data in Motion on the HDP and HDF Platforms. 
             </div> 
             <div class="kbentry-body node-body span11"> 
              <h4>Article</h4> 
              <p><strong>Introduction</strong></p> 
              <p>For NLP, mostly I want to do two things,</p> 
              <ol> 
               <li><strong>Entity Recognition</strong> (people, facility, organizations, locations, products, events, art, language, groups, dates, time, percent, money, quantity, ordinal and cardinal)</li> 
               <li><strong>Sentiment Analysis</strong></li> 
              </ol> 
              <p>So basically what is it and why don't people like it. These two features are very useful as part of a real-time streaming processing of social, email, logs and semistructured document data. I can use both of these in Twitter ingest via Apache NiFi or Apache Spark. Don't confuse text entity recognition with image recognition that we looked at with TensorFlow previously. You can certainly add that to your flow as well, but that is working with images and not text.</p> 
              <p>My debate with sentiment analysis is do you give numbers, really general terms like Neutral, Negative or Positive. Or do you get more detailed like Stanford CoreNLP which has multiple of each.</p> 
              <p>There are a lot of libraries available for NLP and Sentiment Analysis. The first two decisions are do you want to run JVM programs (good for Hadoop MR, Apache Spark, Apache Storm, Enterprise applications, Spring applications, Microservices, NiFi Processors, Hive UDFs, Pig UDFs and have multiple programming language support (Java, Scala, ...). Or run on Python which is already well-known by many data scientists and engineers, is simple to prototype with no compiling, very easy to call from NiFi and scripts and also has a ton of great Deep Learning libraries and interfaces.</p> 
              <p><strong>Python Libraries</strong></p> 
              <p>Like most things in Python, you can use Pip to install them. You will need a Python 2.7 or 3.0 environment setup with PIP to install and use the libraries I have looked at. spaCY requires numpy and so do many of the others.</p> 
              <p><strong>spaCy</strong></p> 
              <pre>pip install -U spacy

              <p><strong>NLTK</strong> does sentiment analysis very easily as shown above. It runs fairly quickly so you can call this in a stream without too much overhead.</p> 
              <p><strong>TextBlob</strong></p> 
              <pre>from textblob import TextBlob

              <p><strong>TextBlob</strong> is a nice library that does Sentiment Analysis as well as spell checking and other useful text processing.</p> 
              <p>The install will look familiar.</p>

              <p><strong>StanfordNLP</strong></p> 
              <p>I love StanfordNLP, it works very well, integrates in a Twitter processing flow and is very accurate. The only issue for many is that it is GPLd and for many use cases will require purchasing a license. It is very easy to use Stanford CoreNLP with Java, Scala and Spark.</p> 
              <pre>import java.util.Properties

import org.apache.spark.SparkContext
import org.apache.spark.sql.SQLContext
import org.apache.phoenix.spark._

import com.vader.SentimentAnalyzer
import edu.stanford.nlp.ling.CoreAnnotations
import edu.stanford.nlp.neural.rnn.RNNCoreAnnotations
import edu.stanford.nlp.pipeline.StanfordCoreNLP
import edu.stanford.nlp.sentiment.SentimentCoreAnnotations
import org.apache.log4j.{Level, Logger}
import org.apache.spark.{SparkConf, SparkContext}
import org.apache.spark.serializer.KryoSerializer
import org.apache.spark.sql._
import scala.collection.JavaConversions._
import scala.collection.mutable.ListBuffer

case class Tweet(coordinates: String, geo:String, handle: String, hashtags: String, language: String,
                 location: String, msg: String, time: String, tweet_id: String, unixtime: String, user_name: String, tag: String, profile_image_url: String,
                 source: String, place: String, friends_count: String, followers_count: String, retweet_count: String, 
                 time_zone: String, sentiment: String, stanfordSentiment: String)

val message = convert(anyMessage)
  val pipeline = new StanfordCoreNLP(nlpProps)
  val annotation = pipeline.process(message)
  var sentiments: ListBuffer[Double] = ListBuffer()
  var sizes: ListBuffer[Int] = ListBuffer()

  var longest = 0
  var mainSentiment = 0

  for (sentence &lt;- annotation.get(classOf[CoreAnnotations.SentencesAnnotation])) {
    val tree = sentence.get(classOf[SentimentCoreAnnotations.AnnotatedTree])
    val sentiment = RNNCoreAnnotations.getPredictedClass(tree)
    val partText = sentence.toString

    if (partText.length() &gt; longest) {
      mainSentiment = sentiment
      longest = partText.length()
    }

    sentiments += sentiment.toDouble
    sizes += partText.length
  }

  val averageSentiment:Double = {
    if(sentiments.nonEmpty) sentiments.sum / sentiments.size
    else -1
  }

  val weightedSentiments = (sentiments, sizes).zipped.map((sentiment, size) =&gt; sentiment * size)
  var weightedSentiment = weightedSentiments.sum / (sizes.fold(0)(_ + _))

  if(sentiments.isEmpty) {
    mainSentiment = -1
    weightedSentiment = -1
  }

  weightedSentiment match {
    case s if s &lt;= 0.0 =&gt; NOT_UNDERSTOOD
    case s if s &lt; 1.0 =&gt; VERY_NEGATIVE
    case s if s &lt; 2.0 =&gt; NEGATIVE
    case s if s &lt; 3.0 =&gt; NEUTRAL
    case s if s &lt; 4.0 =&gt; POSITIVE
    case s if s &lt; 5.0 =&gt; VERY_POSITIVE
    case s if s &gt; 5.0 =&gt; NOT_UNDERSTOOD
  }
}

trait SENTIMENT_TYPE
case object VERY_NEGATIVE extends SENTIMENT_TYPE
case object NEGATIVE extends SENTIMENT_TYPE
case object NEUTRAL extends SENTIMENT_TYPE
case object POSITIVE extends SENTIMENT_TYPE
case object VERY_POSITIVE extends SENTIMENT_TYPE
case object NOT_UNDERSTOOD extends SENTIMENT_TYPE

$(function($) {
    var toplevel = 76935;
    var widget = $('#kbnavigation');

    var updateTree = function(parentId, children) {
        $.ajax({
            type: "GET",
            url: "/kb/reorder/"+parentId+".json",
            data: {nodes: children.join(',')},
            dataType: 'json'
        });
    };

    var onChange = function(event, data) {
        var parentLi = data.item.parents('li')[0],
            newLi = data.item[0],
            ul = newLi.parentElement,
            elems = ul.children,
            parentId = toplevel,
            sortElems = [];
        if (parentLi) {
            parentId = parentLi.getAttribute('data-id');
        }
        for(var i = 0, l = elems.length; i < l; i++) {
            sortElems.push(elems[i].getAttribute('data-id'));
        }
        updateTree(parentId, sortElems);
    };

    var nav = widget.find('ul').nestedSortable({
        handle: 'a',
        listType: 'ul',
        items: 'li',
        toleranceElement: '> a',
        relocate: function(arg1, arg2) {
            onChange(arg1, arg2);
            makeNavigation(widget);
        }
    });

    var toggleCaret = function(caret) {
        var $i = $(caret);
        var $ul = $i.parent().find('ul').first();
        if ($i.hasClass('icon-caret-right')){
            $i.removeClass('icon-caret-right').addClass('icon-caret-down');
            $ul.slideDown('fast');
        } else if ($i.hasClass('icon-caret-down')){
            $i.removeClass('icon-caret-down').addClass('icon-caret-right');
            $ul.slideUp('fast', function(){
                //Fix for Chrome bug when collapsing
                $i.next().show(function(){});
            });
        }
    };

    var addCaret = function(el, first) {
        if (el.children.length > 0 && el.children[0].nodeName === 'I') return;
        var $li = $(el);
        var i = document.createElement('I');
        var $caret = $(i).on('click', function() {
            toggleCaret(this);
        });
        if (!$li.hasClass('active') && first === true) {
            $li.prepend($caret.addClass('icon-caret-right'));
        } else {
            $li.prepend($caret.addClass('icon-caret-down'));
        }
    };

    var removeCaret = function(el) {
        if (el.children.length === 0 || el.children[0].nodeName !== 'I') return;
        el.removeChild(el.children[0]);
    };

    var makeNavigation = function(widget, first) {
        $li = widget.find('li');
        for (var i = 0, len = $li.length; i < len; i += 1) {
            var li = $li[i];
            var ul = li.getElementsByTagName('UL');
            if (ul.length === 0) {
                removeCaret(li);
                continue;
            }
            var childLi = ul[0].getElementsByTagName('li');
            if (childLi.length === 0) {
                removeCaret(li);
                continue;
            }
            addCaret(li, first);
        }

        if (first === true) {
            // hide all
            widget.find('li > ul').toggle();

            // show child articles
            widget.find('.active > ul').toggle();

            // show all parents
            var active = widget.find('li.active');

            var parents = active.parents('ul, li');
            for (var i = 0, l = parents.length; i < l; i++) {
                if (parents[i].nodeName === 'UL') {
                    var ul = $(parents[i]);
                    if (ul.hasClass('navigation-top')) {
                        break;
                    }
                    ul.toggle();
                } else if (parents[i].nodeName === 'LI') {
                    var li = $(parents[i]);
                    li.addClass('open');
                    li.find('i').first().removeClass('icon-caret-right').addClass('icon-caret-down');
                }
            }
        }

    };

    makeNavigation(widget, true);

});
