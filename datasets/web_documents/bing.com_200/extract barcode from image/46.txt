       <h2 class="text_red">MICR Reader SDK</h2> 
       <p>Read MICR lines from image of checks stored image image or Image Cash Letter (X9.37) files. Obtain Routing, Account, Check number, Amount information. Extract image of a check from scanned or faxed documents.</p>
       <br> Just a few code lines extract MICR data from your image 
       <a href="#cs" class="btn_code fancybox" title="C#">C#</a> 
       <a href="#vb" class="btn_code fancybox" title="VB">VB</a> 
       <a href="#cpp" class="btn_code fancybox" title="C++">C++</a> 
       <a href="#php" class="btn_code fancybox" title="PHP">PHP</a> 
       <a href="#asp" class="btn_code fancybox" title="ASP / VBScript">VBScript</a> See 
       <a href="http://how-to.inliteresearch.com/micr-line-reading-howto/" target="_blank">more examples</a> 
       <!-- see https://google-code-prettify.googlecode.com/svn/trunk/README.html --> 
       <div id="cs" style="display: none;">
        <pre class="prettyprint lang-cs">
ClearMicr.CcMicrReader reader = new ClearMicr.CcMicrReader();
reader.Image.Open("myFile.tif", 1);
int cnt = reader.FindMICR();
if (cnt &gt; 0) {
   Console.WriteLine("ROUTING: " +  reader.MicrLine[1].Routing.TextANSI +  reader.MicrLine[1].RoutingChecksum.TextANSI);
   Console.WriteLine("ACCOUNT: " +  reader.MicrLine[1].Account.TextANSI); }
</pre>
       </div> 
       <div id="vb" style="display: none;">
        <pre class="prettyprint lang-vb">
Dim reader As New ClearMicr.CcMicrReader()
reader.Image.Open("myFile.tif", 1)
Dim cnt As Integer = reader.FindMICR()
If cnt &gt; 0 Then
	Console.WriteLine("ROUTING: " + reader.MicrLine(1).Routing.TextANSI + reader.MicrLine(1).RoutingChecksum.TextANSI)
	Console.WriteLine("ACCOUNT: " + reader.MicrLine(1).Account.TextANSI)
End If
</pre>
       </div>
