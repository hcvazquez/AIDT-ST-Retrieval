        function getCookie(name) {
            var nameEQ = name + '=';
            var ca = document.cookie.split(';');
            for(var i=0;i < ca.length;i++) {
                var c = ca[i];
                while (c.charAt(0)==' ') c = c.substring(1,c.length);
                if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
            }

            return null;
        }

        function set(name, value) {
            if (hasSessionStorage()) {
                sessionStorage.setItem(name, value);
            } else {
                setCookie(name, value);
            }
        }

        function setCookie(name, value, expiryInDays, domain) {
            var expires = '';
            expiryInDays = parseInt(expiryInDays) || 0;

            if (!domain) {
                domain = document.domain
                    .split('.')
                    .slice(-2)
                    .join('.');

                domain = '.' + domain;
            }

            if (expiryInDays) {
                var date = new Date();
                date.setDate(date.getDate() + expiryInDays);
                expires = '; expires=' + date.toUTCString();
            }

            domain = '; domain=' + domain;
            document.cookie = name + '=' + value + domain + expires + '; path=/';
        }

        function hasSessionStorage() {
            var testStr = 'dl';
            try {
                sessionStorage.setItem(testStr, testStr);
                sessionStorage.removeItem(testStr);
                return true;
            } catch(e) {
                return false;
            }
        }

        function getOs() {
            var appVersion = window.navigator.appVersion;
            var osMap = [
                { os: 'windows', re: /windows/i },
                { os: 'ios', re: /iphone|ipad|ipod/i },
                { os: 'android', re: /android/i },
                { os: 'mac', re: /mac/i }
            ];

            for (var i = 0; i < osMap.length; i++) {

                    return osMap[i].os;
                }

                        function medusa_dfp_listener(e){
                            if (e.data.match && e.data.match(/sender.*leadgenads/g) ) {
                try {
                    var data = JSON.parse( e.data );
                } catch (e) {
                    console.log("leadgenads can't be parsed to json");
                    return;                
                }

                DL.medusaTrackerVars['promo'] = data['promo'];                          DL.medusaTrackerVars['clickurl'] = data['clickurl'];    
                var count = 0;                                      
                var interval = setInterval(function(){
                    if(count >= 7) {
                        clearInterval(interval);
                        console.log("Unable to make medusa call due to jQuery not loading within " + count + " seconds.");
                        return;
                    } 

                    if( jQueryLoaded() ) {
                        clearInterval(interval);
                        makeMedusaCall(data);
                    } else {
                        count++;
                    }
                },1000);
              }

            function jQueryLoaded() {
                return (typeof $ == 'undefined') ? false : true;

            function makeMedusaCall(data) {
                $.ajax({
                    url: '/medusa/recommendation/'+data['spotname']+'/',
                    data: DL.medusaTrackerVars,
                    type: 'GET',
                    success: function(d) {
                        if(d.success == true) {
                            $("."+data["divLabel"]).html(d.html);
                            initMedusaTracking(data);
                            dispatchLoadedEvent();
                        }
                    }
                });   
            }

            function initMedusaTracking(data) {
                                var $medusaButtons = $('.' + data['divLabel']).find("a#medusaButton");
                var $docIds = "";
                if ($medusaButtons.length > 0) {
                    var $docIdArr = [];
                    $medusaButtons.each(function() {
                        var $idValue = $(this).data('docid') + '';
                        console.log("$idValue: " + $idValue);
                        if ($idValue.length > 0) {
                            $docIdArr.push($idValue);
                        }
                    });

                    if ($docIdArr.length > 0) {
                        $docIds = $docIdArr.join(',');
                    } else {
                        console.log("Ad units returned for spot: " + data['spotname'] + ", but none are Medusa.");
                        return;
                    }
                } else {
                    console.log("No ad units returned for spot: " + data['spotname']);
                    return;
                }

                DL.rdsTrackingVars['spotname'] = data['spotname'];
                DL.rdsTrackingVars['promotionId'] = data['promo'];
                DL.rdsTrackingVars['ReferringURL'] = $(location).attr("href");
                DL.rdsTrackingVars['docId'] = $docIds;

                                $.ajax({
                    url: '/module/rds/impression/',
                    data: DL.rdsTrackingVars,
                    type: 'GET'
                });

                                var $medusaBtnEnabled = true;
                $medusaButtons.each(function() {
                    var $medusaBtn = $(this);
                    $medusaBtn.on('click', function(e) {
                        e.preventDefault();

                        if ($medusaBtnEnabled) {
                            $medusaBtnEnabled = false;

                                                        DL.rdsTrackingVars['docId'] = $medusaBtn.data('docid');

                            $.ajax({
                                url: '/module/rds/click/',
                                data: DL.rdsTrackingVars,
                                type: 'GET',
                                success: function() {
                                    $medusaBtnEnabled = true;
                                    window.location = $medusaBtn.attr('href');
                                }
                            });
                        }
                    })
                });
            }

            function readMedusaCookie(name) {
                var nameEQ = name + "=";
                var ca = document.cookie.split(';');
                for(var i=0;i < ca.length;i++) {
                    var c = ca[i];
                    while (c.charAt(0)==' ') c = c.substring(1,c.length);
                    if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
                }
                return null;
            }

            function dispatchLoadedEvent() {
                var event = document.createEvent('CustomEvent');
                event.initCustomEvent('medusaUnitLoaded', true, false, {});
                document.documentElement.dispatchEvent(event);
            }
