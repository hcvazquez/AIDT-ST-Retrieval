        var weatherElement = document.querySelector('#current-weather');
        var currentWeather;                
        
        function getCurrentWeather () {
            var current;
            var xhr = new XMLHttpRequest();
            xhr.open('POST', '/Service?method=Weather.getPB');
            
            // Listen for request completion
            xhr.addEventListener('load', function () {
                // Parse the response
                var response = JSON.parse(this.responseText);
                
                // Get the current weather
                var current = {
                    summary: response.currently.summary.toLowerCase(),
                    icon: response.currently.icon,
                    timestamp: Date.now()
                };
                
                // Update the weather element text
                weatherElement.textContent = current.summary;
                
                // Save to local storage
                localStorage.currentWeather = JSON.stringify(current);
            });
            
            // Send the request
            xhr.send();
        }
        
        function displayWeather () {
            if (localStorage.currentWeather) {
                // Parse saved current weather
                currentWeather = JSON.parse(localStorage.currentWeather);
                
                // Check if last saved time is over an hour ago
                if (Date.now() - currentWeather.timestamp > 60000) {
                    // Request new weather
                    getCurrentWeather();                
                } else {
                    // Update the weather element
                    weatherElement.textContent = currentWeather.summary;                    
                }
            } else {
                getCurrentWeather();
            }
        }

            displayWeather();
        } catch (err) {
            // Local storage not supported to just say it's sunny
