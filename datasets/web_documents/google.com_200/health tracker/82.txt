	function performBrowserChecks() {

	function cookiesEnabled() {
		var cookieEnabled = navigator.cookieEnabled;

		//if navigator,cookieEnabled is not supported
		if (typeof navigator.cookieEnabled == "undefined" && !cookieEnabled) {
			document.cookie = "testcookie";
			cookieEnabled   = document.cookie.indexOf("testcookie") != -1;
		}

		return cookieEnabled;
	}

	function bindAjaxHandlers() {
		$(document).bind(
				'ajaxError',
				function (event, request, settings) {
					var jsonResult = null;
					try {
						jsonResult = jQuery.parseJSON(request.responseText);
					} catch (error) {}

					if (jsonResult != null && jsonResult != undefined) {
						// ajax call session expiration redirect
						if (request.status == 403 && jsonResult.wsrStatusCode === 'SESSION_EXPIRED') {
							location.replace('/');
						} else if (request.status == 307 && jsonResult.wsrStatusCode === 'SITE_MAINTENANCE') {
							window.location = jsonResult.wsrStatusMessage;
						} else if (!mAjax_generalMsgOverideEnabled) {
							displayAjaxGeneralMessage(mAjax_generalError, mAjax_generalErrorTitle);
						}
					} else {
						// add appropriate redirection code here
						if (!mAjax_generalMsgOverideEnabled) {
							displayAjaxGeneralMessage(mAjax_generalError, mAjax_generalErrorTitle);
						}
					}
				});
	}
