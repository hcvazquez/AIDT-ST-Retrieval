         <p>This is a code snippet for basic JavaScript to validate email address using a regular expression. This is building on my previous post on <a href="http://www.jquery4u.com/javascript/basic-javascript-regular-expression/">how to use regular expressions with jQuery</a>. You can also load the code in jsfiddle below.</p> 
         <p><strong>Update 12/05/13:</strong> Separated into versions for testing.</p> 
         <h2>Version 1</h2> 
         <p><iframe width="100%" height="80" src="https://jsfiddle.net/jquery4u/5rPmV/show/" allowfullscreen frameborder="0"></iframe></p> 
         <pre><code class="language-javascript">var $email = $('form input[name="email'); //change form to id or containment selector
var re = /^(([^&lt;&gt;()[]\.,;:s@"]+(.[^&lt;&gt;()[]\.,;:s@"]+)*)|(
".+"))@(([[0-9]{1,3}.[0-9]{1,3}.[0-9]{1,3}.[0-9]{1,3}])|(([a-zA
-Z-0-9]+.)+[a-zA-Z]{2,}))$/igm;
if ($email.val() == '' || !re.test($email.val()))
{
    alert('Please enter a valid email address.');
    return false;
}</code></pre> 
         <p><a class="seelivedemo" href="https://jsfiddle.net/jquery4u/5rPmV/">Load code in jsfiddle</a></p> 
         <h2>Version 2</h2> 
         <p><iframe width="100%" height="80" src="https://jsfiddle.net/jquery4u/T2uLp/show/" allowfullscreen frameborder="0"></iframe></p> 
         <pre><code class="language-javascript">var $email = $('form input[name="email'); //change form to id or containment selector
var re = /[A-Z0-9._%+-]+@[A-Z0-9.-]+.[A-Z]{2,4}/igm;
if ($email.val() == '' || !re.test($email.val()))
{
    alert('Please enter a valid email address.');
    return false;
}</code></pre>
