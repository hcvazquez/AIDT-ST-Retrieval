      $("#include_personal_glossary").click(function() {
        $.ajax({
          type: "POST",
          url: "/ajax/ajax_kudoz.php",
          dataType: "json",
          data: {
            action: "get_personal_glossaries",
            eid_v: "0"
          },
          success: function(json) {
            if (json.is_success) {
              var error = false;
              $("#personal_glossaries_list").html(json.data).show();
              $("#include_button").click(function() {
                if ($("#list_select").val() == "") {
                  alert("Select a glossary");
                  error = true;
                }
                if ($("#list_select").val() == "new") {
                  glossary_name = $("#new_glossary_name").val();
                  glossary_id = 0;
                  if ($.trim(glossary_name) == "") {
                    alert("Insert a glossary name");
                    error = true;
                  }
                } else {
                  glossary_id = $("#list_select").val();
                  glossary_name = "";
                }
                if (!error) {
                  $.ajax({
                    type: "POST",
                    url: "/ajax/ajax_kudoz.php",
                    dataType: "json",
                    data: {
                      action: "add_term_to_personal_glossary",
                      eid_v: "0",
                      term_id: "3029554",
                      glossary_id: glossary_id,
                      glossary_name: glossary_name
                    },
                    success: function(status) {
                      if (status.is_success) {
                        $("#kog_personal_glossary").html("<a href='/?sp=gloss/term&id=" + status.is_success + "'>Already in personal glossary</a>");
                        alert("Term included in your personal glossary");
                      } else {
                        alert("Error including this term on your personal glossary, pleaee try again");
                      }
                      $("#personal_glossaries_list").html("");
                    }
                  });
                }
              });
              $("#cancel_button").click(function() {
                $("#personal_glossaries_list").html("").hide();
              });
              $("#list_select").change(function() {
                if ($("#list_select").val() == "new") {
                  $("#new_glossary_name").show();
                } else {
                  $("#new_glossary_name").hide();
                }
              });
            } else {
              alert("Error getting your glossaries list");
            }
          }
        });
      });
