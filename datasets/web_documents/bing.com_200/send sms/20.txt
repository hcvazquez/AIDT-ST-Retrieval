             <pre lang="Java"><span class="code-keyword">package</span> com.example.androidreadsms;

<span class="code-keyword">import</span> android.app.Activity;
<span class="code-keyword">import</span> android.database.Cursor;
<span class="code-keyword">import</span> android.net.Uri;
<span class="code-keyword">import</span> android.os.Bundle;
<span class="code-keyword">import</span> android.view.Gravity;
<span class="code-keyword">import</span> android.view.Menu;
<span class="code-keyword">import</span> android.view.MenuItem;
<span class="code-keyword">import</span> android.view.View;
<span class="code-keyword">import</span> android.view.View.OnClickListener;
<span class="code-keyword">import</span> android.widget.Button;
<span class="code-keyword">import</span> android.widget.TableLayout;
<span class="code-keyword">import</span> android.widget.TableRow;
<span class="code-keyword">import</span> android.widget.TextView;
<span class="code-keyword">import</span> android.widget.Toast;

<span class="code-keyword">public</span> <span class="code-keyword">class</span> MainActivity <span class="code-keyword">extends</span> Activity
    <span class="code-keyword">implements</span> OnClickListener {

    Button btnAll;
    Button btnInbox;
    Button btnSent;
    Button btnDraft;
    TableLayout tblMain;
    
    <span class="code-sdkkeyword">@Override</span>
    <span class="code-keyword">protected</span> <span class="code-keyword">void</span> onCreate(Bundle savedInstanceState) {
        <span class="code-keyword">super</span>.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        <span class="code-keyword">try</span> {
            <span class="code-comment">/*</span><span class="code-comment">
             * Initializing Widgets
             */</span>

            btnAll = (Button) findViewById(R.id.btnAll);
            btnAll.setOnClickListener(<span class="code-keyword">this</span>);
            
            btnInbox = (Button) findViewById(R.id.btnInbox);
            btnInbox.setOnClickListener(<span class="code-keyword">this</span>);
            
            btnSent = (Button) findViewById(R.id.btnSent);
            btnSent.setOnClickListener(<span class="code-keyword">this</span>);
            
            btnDraft = (Button) findViewById(R.id.btnDraft);
            btnDraft.setOnClickListener(<span class="code-keyword">this</span>);
            
            tblMain = (TableLayout) findViewById(R.id.tblMain);
            
        } <span class="code-keyword">catch</span> (Exception ex) {
            Toast.makeText(<span class="code-keyword">this</span>,
                    <span class="code-string">"</span><span class="code-string">Error in MainActivity.onCreate: "</span> + ex.getMessage(),
                    Toast.LENGTH_SHORT).show();
        }
    }

    <span class="code-sdkkeyword">@Override</span>
    <span class="code-keyword">public</span> <span class="code-keyword">void</span> onClick(View v) {
        Uri smsUri = Uri.parse(<span class="code-string">"</span><span class="code-string">content://sms/"</span>);
        
        <span class="code-keyword">switch</span> (v.getId()) {
        <span class="code-keyword">case</span> R.id.btnInbox:
            smsUri = Uri.parse(<span class="code-string">"</span><span class="code-string">content://sms/inbox"</span>);
            <span class="code-keyword">break</span>;
        <span class="code-keyword">case</span> R.id.btnSent:
            smsUri = Uri.parse(<span class="code-string">"</span><span class="code-string">content://sms/sent"</span>);
            <span class="code-keyword">break</span>;
        <span class="code-keyword">case</span> R.id.btnDraft:
            smsUri = Uri.parse(<span class="code-string">"</span><span class="code-string">content://sms/draft"</span>);
            <span class="code-keyword">break</span>;
        }
        
        Cursor cursor = getContentResolver().query(smsUri, null, null, null, null);

        Cursor2TableLayout(cursor, tblMain);
    }
    
    <span class="code-keyword">public</span> <span class="code-keyword">void</span> Cursor2TableLayout(Cursor cur, TableLayout tblLayout) {
        
        <span class="code-comment">/*</span><span class="code-comment"> Clearing Table If Contains Any Rows/Headers */</span>
        tblLayout.removeAllViews();
        
        <span class="code-comment">/*</span><span class="code-comment"> Moving To First */</span>
        <span class="code-keyword">if</span> (!cur.moveToFirst()) { <span class="code-comment">/*</span><span class="code-comment"> false = cursor is empty */</span>
            <span class="code-keyword">return</span>;
        }

        <span class="code-comment">/*</span><span class="code-comment"> Column Headers */</span>
        
        TableRow headersRow = <span class="code-keyword">new</span> TableRow(<span class="code-keyword">this</span>);
        
        <span class="code-keyword">for</span> (<span class="code-keyword">int</span> j = <span class="code-digit">0</span>; j &lt; cur.getColumnCount(); j++) {
            TextView textView = <span class="code-keyword">new</span> TextView(<span class="code-keyword">this</span>);

            textView.setGravity(Gravity.CENTER_HORIZONTAL);
            
            textView.setText(cur.getColumnName(j));

            textView.setPadding(<span class="code-digit">0</span>, <span class="code-digit">0</span>, <span class="code-digit">5</span>, <span class="code-digit">0</span>);
            
            <span class="code-keyword">if</span> (Build.VERSION.SDK_INT &gt;= <span class="code-digit">11</span>) { <span class="code-comment">/*</span><span class="code-comment"> If running Android 3.0+ */</span>
                textView.setAlpha(<span class="code-digit">0</span>.8f);
            } <span class="code-keyword">else</span> {
                <span class="code-comment">/*</span><span class="code-comment"> Phones with Android 2.x are a rarity now,
                 * but until recently (June, 2015) I have seen them
                 * in the store...
                 * Also, I myself have such gadget and using it sometimes... */</span>
                AlphaAnimation animation = <span class="code-keyword">new</span> AlphaAnimation(<span class="code-digit">0</span>.8f, <span class="code-digit">0</span>.8f);
                animation.setDuration(<span class="code-digit">0</span>);
                animation.setFillAfter(true);
                textView.startAnimation(animation);
            }
        
            headersRow.addView(textView);
        }

        headersRow.setPadding(<span class="code-digit">10</span>, <span class="code-digit">10</span>, <span class="code-digit">10</span>, <span class="code-digit">10</span>);

        tblLayout.addView(headersRow);
        
        <span class="code-comment">/*</span><span class="code-comment"> Rows */</span>
    
        <span class="code-keyword">for</span> (<span class="code-keyword">int</span> i = <span class="code-digit">0</span>; i &lt; cur.getCount(); i++) {

            TableRow tableRow = <span class="code-keyword">new</span> TableRow(<span class="code-keyword">this</span>);

            <span class="code-keyword">for</span> (<span class="code-keyword">int</span> j = <span class="code-digit">0</span>; j &lt; cur.getColumnCount(); j++) {
                    TextView textView = <span class="code-keyword">new</span> TextView(<span class="code-keyword">this</span>);

                    textView.setGravity(Gravity.CENTER_HORIZONTAL);

                    textView.setText(cur.getString(j));

                    textView.setPadding(<span class="code-digit">0</span>, <span class="code-digit">0</span>, <span class="code-digit">5</span>, <span class="code-digit">0</span>);
                
                    tableRow.addView(textView);
            }

            tableRow.setPadding(<span class="code-digit">10</span>, <span class="code-digit">10</span>, <span class="code-digit">10</span>, <span class="code-digit">10</span>);

            tblLayout.addView(tableRow);
            cur.moveToNext();
        }

        cur.close();
    }
