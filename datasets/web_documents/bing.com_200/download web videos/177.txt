	var langs = $.parseJSON(langs);		
	if(currLang == "") currLang = "en-us";
	
	$.each(langs, function(key, item) {
		 if(key == currLang){
			 $("#currLang").prepend(item);
			 $("#currLang").attr("href", "?lang="+key);
			 $("#footerLangList").append('<option value="?lang='+key+'" selected>' + item + '</option>');
		 }else{
			$("#headerLangList").append('<a href="?lang='+key+'">' + item + '</a>');
			$("#footerLangList").append('<option value="?lang='+key+'">' + item + '</option>');
		 }
	 });
}
