    function selectPerson(index) {
        var img, i, url;
        if (selected_person >= 0) {
            img = document.getElementById("person_" + selected_person);
            if (img) {
                img.style.opacity = "0.4"; 
                img.style.filter = "alpha(opacity=40)";
            }
        }
        img = document.getElementById("person_" + index);
        if (img) {
            selected_person = index;
            img.style.opacity = "1.0"; 
            img.style.filter = "alpha(opacity=100)";
            flyover = document.getElementById('related_item_flyover');
            flyoverContents = flyoverContentsFor(img);
            innerHTML = '<span class="person-name"><a href="author_page.cfm?id=' + id + '" onclick="feature(5)" target="_new">' + flyoverContents[0] + '</a></span><br/>';
            innerHTML += '<table class="person-concepts">';
            for (i = 0; i < flyoverContents[2].length; i++) {
                if (!i % 2) {
                    innerHTML += '<tr>';
                }
                url = flyoverContents[3][i];
                title = flyoverContents[4][i];
                if (url) {
                    innerHTML += '<td width="250px"><span class="person-concept" onmouseenter="personConceptEnter(' + index + ',' + i + ')"><a href="' + url +  '"' + ' onclick="feature(4)" '  + ' title="' + 'Appears in paper: ' + flyoverContents[4][i] + '" ' + 'target="_new">'+ flyoverContents[2][i] + '</a></span></td>';
                }
                else {
                    innerHTML += '<td width="250px"><span class="person-concept" onmouseenter="personConceptEnter(' + index + ',' + i + ')">' + flyoverContents[2][i] + '</span></td>';
                }
                if (i % 2) {
                    innerHTML += '</tr>';
                }
            }
            if (i % 2) {
                innerHTML += '<td></td></tr>';
            }
            innerHTML += '</table>';
            innerHTML += '<br/>';
            flyover.innerHTML = innerHTML;
            flyover.style.opacity = 8.0;
            flyover.style.filter = "alpha(opacity=80)";  // for ie
        }
    }

    function flyoverContentsFor(img) {
        var index, i, name, tags = [], urls = []; titles = [];
        index = img.id.split('_')[1];
        name = related_authors[index].name;
        id = related_authors[index].id;
        for (i = 0; i < related_authors[index].concepts.length; i++) {
            tags[i] = related_authors[index].concepts[i].label;
            urls[i] = related_authors[index].concepts[i].url;
			titles[i] = related_authors[index].concepts[i].source;
        }
        return ([name, id, tags, urls, titles]);
    }

    function relatedItemEnter(img) {
        document.getElementById('person_concept_explanation').innerHTML = '';
        selectPerson(img.id.split('_')[1]);
		personConceptEnter(img.id.split('_')[1],0); //wg
    }
	
    function personConceptEnter(personIndex, conceptIndex) {

			source + '</a></div>' + '<span class="person-concept-author">' + related_authors[personIndex].name + ' wrote:<br/></span><span class="person-concept-passage">&quot;' + passage + '&quot;</span>';
            
			
			div.innerHTML = innerHTML;
        }
        else {
            div.innerHTML = '';
        }
    }

         <h2><span style="white-space: nowrap;">Concepts in</span><span>Certification of Algorithms 63, 64, 65: Partition, quicksort, find</span></h2> 
         <dl> 
          <dt style="line-height: 200%;">
           <strong>Quicksort</strong>
          </dt> 
          <dd>
           Quicksort is a sorting algorithm developed by Tony Hoare that, on average, makes comparisons to sort n items. In the worst case, it makes comparisons, though this behavior is rare. Quicksort is often faster in practice than other algorithms. Additionally, quicksort's sequential and localized memory references work well with a cache. Quicksort can be implemented with an in-place partitioning algorithm, so the entire sort can be done with only additional space.&nbsp;
