        <h1>Credit Card Validation using LUHN formula</h1>
        <div style="padding-bottom:0px; margin-top:-10px; height:24px;">
         <g:plusone size="medium" href="http://freevbcode.com/ShowCode.asp?ID=3132" annotation="inline"></g:plusone>

         <p><strong>Version Compatibility:</strong><em> Visual Basic 6, Visual Basic 5 </em></p> 
         <div class="longTitle">
          More information: 
         </div> 
         <div id="body">
           This funciton will validate credit cards using the LUHN formula and, unlike many other examples, it also checks that the length of the number is valid dependant on the card issuer.
         </div> 
         <p><strong>Instructions:</strong> Copy the declarations and code below and paste directly into your VB project.</p> 
         <p><strong>Declarations:</strong><br> <textarea cols="65" rows="6">' (None)</textarea> </p> 
         <strong>Code:</strong>
         <br> 
         <textarea cols="65" rows="20">Public Function ValidCard(ByVal CardNumber As String) As Boolean

Dim intCount As Integer
Dim intValue As Integer
Dim intArr() As Integer
Dim intStart As Integer
Dim intArrValue As Integer

    ValidCard = False
    
    If Not IsNumeric(CardNumber) Then Exit Function
    
    If Left(CardNumber, 4) = 2131 Or Left(CardNumber, 4) _
       = 1800 Then
        'JCB
        If Len(CardNumber) &lt;&gt; 15 Then Exit Function
    ElseIf Left(CardNumber, 4) = 2149 Or _
          Left(CardNumber, 4) = 2014 Then
        'enroute
        If Len(CardNumber) = 15 Then
            ValidCard = True
        End If
        Exit Function
    ElseIf Left(CardNumber, 4) = 6011 Then
        'discover
        If Len(CardNumber) &lt;&gt; 16 Then Exit Function
    ElseIf (Left(CardNumber, 3) &gt;= 300 And _
        Left(CardNumber, 3) &lt;= 305) Or _
        Left(CardNumber, 2) = 36 Or Left(CardNumber, 2) = 38 Then
        'diners club
        If Len(CardNumber) &lt;&gt; 14 Then Exit Function
    ElseIf Left(CardNumber, 2) = 34 Or _
          Left(CardNumber, 2) = 37 Then
        'amex
        If Len(CardNumber) &lt;&gt; 15 Then Exit Function
    ElseIf Left(CardNumber, 2) &gt;= 51 And _
           Left(CardNumber, 2) &lt;= 55 Then
        'mastercard
        If Len(CardNumber) &lt;&gt; 16 Then Exit Function
    ElseIf Left(CardNumber, 1) = 4 Then
        'visa
        If Not (Len(CardNumber) = 13 Or Len(CardNumber) = 16) _
            Then Exit Function
    ElseIf Left(CardNumber, 1) = 3 Then
        'jcb (2)
        If Len(CardNumber) &lt;&gt; 16 Then Exit Function
    Else
        'unknown card.....
        Exit Function
    End If
    ReDim intArr(Len(CardNumber))
    For intCount = Len(CardNumber) - 1 To 1 Step -2
        intValue = Mid(CardNumber, intCount, 1) * 2
        intArr(intCount) = intValue
    Next intCount
    
    intValue = 0
    
    If Len(CardNumber) Mod 2 = 0 Then
        intStart = 2
    Else
        intStart = 1
    End If
    
    For intCount = intStart To Len(CardNumber) Step 2
        intValue = intValue + Mid(CardNumber, intCount, 1)
        intArrValue = intArr(intCount - 1)
        If intArrValue &lt; 10 Then
            intValue = intValue + intArrValue
        Else
            intValue = intValue + Left(intArrValue, 1) + _
              Right(intArrValue, 1)
        End If
    Next intCount
    
    If intValue Mod 10 &lt;&gt; 0 Then
        Exit Function
    Else
        ValidCard = True
    End If
        
End Function</textarea>
