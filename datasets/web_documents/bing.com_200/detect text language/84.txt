(function() {
// Check if cookies are enabled
function cookiesEnabled() {
return $.cookie('check', 'valid', { expires: 1 }) && $.cookie('check') == 'valid';
}
// Set the language of the domain and redirect to the domain
function setDomainLanguage(language) {
if($.cookie("lang") != null) {
$.cookie("lang", null, { path: '/', domain: 'ch.mathworks.com'});
}
$.cookie("lang", 'lang=' + language, {path: '/', expires: 365, domain: 'ch.mathworks.com'});
}
// do not display the domain selector if querystring is not present
// or if query string has nocookie value
// or if cookies are disabled
var displayDomainSelector = true;
function detectDomain() {
var queries = {};
$.each(document.location.search.substr(1).split('&'), function(c,q){
if(q != undefined && q != "") {
var i = q.split('=');
if(i[1] != undefined) {
queries[i[0].toString()] = i[1].toString();
}
}
});
var requested_domain = (queries['requestedDomain'] != undefined) ? queries['requestedDomain'] : "";
var nocookie = (queries['nocookie'] != undefined) ? queries['nocookie'] : "";
var client = (queries['client'] != undefined) ? queries['client'] : "";
if(requested_domain == "" || nocookie == "true" || !cookiesEnabled() || client == "addons" || window.overrideDomainSelector) {
displayDomainSelector = false;
}
}
detectDomain();
function extractDomain(url) {
var domain;
//find & remove protocol (http, ftp, etc.) and get domain
if (url.indexOf("://") > -1) {
domain = url.split('/')[2];
} else {
domain = url.split('/')[0];
}
//find & remove port number
domain = domain.split(':')[0];
return domain;
}
function getDomainContent(returnContent) {
// Get the location information from the marketing site
$.ajax({
url : "/includes_content/domainSelect/location.html",
type : "get",
async: false,
success : function(data) {
$("#location_content").html(data);
var button_link = $("#button_link").text();
var current_location = $("#current_location").text();
var location_key = $("#location_key").text();
//Set the location in the text and button
$(".text_location").text(current_location);
$("#domain_select_button").text("Select " + current_location);
// if the current location is different from the browser location, update the button functionality
var browser_domain = extractDomain(window.location.href);
if(button_link.indexOf(browser_domain) == -1) {
// Get the current page link from the links within the selector
new_button_link = $("." + location_key).attr("href");
$("#domain_select_button").attr("href",new_button_link);
$("#domain_select_button").removeAttr("data-dismiss");
}
}
});
if(returnContent) {
return $('#country-unselected').html();
}
}
$(document).ready(function () {
//setLocation();

$("#country-unselected li a").each(function() {
var link_url = $(this).attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");

});
}
if ( $( "li.headernav_login > a" ).length ) {
var login_url = $("li.headernav_login > a").attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");
$("li.headernav_login > a").attr("href", login_url);
}
if ( $( "li.mwa-nav_logout > a" ).length ) {
var login_url = $("li.mwa-nav_logout > a").attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");
$("li.mwa-nav_logout > a").attr("href", login_url);
}
if ( $( ".mwa-nav li a" ).length ) {
$(".mwa-nav li a").each(function() {
var link_url = $(this).attr("href").replace(/%26requestedDomain(%3D[^%26|&]*)?|requestedDomain(%3D[^%26|&]*)?(%26)?/, "");
$(this).attr("href", link_url);
});
}

getDomainContent(false);
});

$('body').on('click','#swiss_english',function(){

});
// set the click function for Switzerland/French link
$('body').on('click','#swiss_french',function(){
setDomainLanguage('fr');
});
// set the click function for the Switzerland/German link
$('body').on('click','#swiss_german',function(){

});
});

if(displayDomainSelector) {
getDomainContent(false);
$('#country-unselected').modal('show');
}
});
})(jQuery);

           <h2 id="bt548t1-3">Input Arguments</h2>
           <div class="expandableContent">
            <p class="switch "><a href="javascript:void(0);" class="expandAllLink" data-allexpanded="true">collapse all</a></p>

             <div class="panel-group" role="tablist" aria-multiselectable="true">
              <div class="panel panel-default " id="bt548t1-1-I">
               <div itemscope itemprop="inputargument" itemtype="http://www.mathworks.com/help/schema/MathWorksDocPage/NamedInputArgument" itemid="input_argument_d119e139805">
                <div role="tab" data-toggle="collapse" id="expand_panel_heading_input_argument_d119e139805" data-target="#expand_panel_body_input_argument_d119e139805" aria-controls="expand_panel_body_input_argument_d119e139805" class="panel-heading add_cursor_pointer" aria-expanded="true">
                 <span id="inputarg_I" class="'anchor_target'"></span>
                 <h3 class="panel-title" id="input_argument_d119e139805"><code itemprop="name">I</code> — <span itemprop="purpose">Input image</span><br> <span class="add_font_color_general remove_bold"><span class="example_desc"><span itemprop="inputvalue"><em class="varname">M</em>-by-<em class="varname">N</em>-by-3 truecolor image</span> | <span itemprop="inputvalue"><em class="varname">M</em>-by-<em class="varname">N</em> 2-D grayscale image</span> | <span itemprop="inputvalue"><em class="varname">M</em>-by-<em class="varname">N</em> binary image</span></span></span></h3>
                 <span class="icon-arrow-open-down icon_24"></span>
                 <span class="icon-arrow-open-right icon_24"></span>
                </div>
                <div role="tabpanel" id="expand_panel_body_input_argument_d119e139805" aria-labelledby="expand_panel_heading_input_argument_d119e139805" class="panel-collapse collapse in">
                 <div class="panel-body"> 
                  <p>Input image, specified in <em class="varname">M</em>-by-<em class="varname">N</em>-by-3 truecolor, <em class="varname">M</em>-by-<em class="varname">N</em> 2-D grayscale, or binary format. The input image must be a real, nonsparse value. The function converts truecolor or grayscale input images to a binary image, before the recognition process. It uses the Otsu’s thresholding technique for the conversion. For best ocr results, the height of a lowercase ‘x’, or comparable character in the input image, must be greater than 20 pixels. From either the horizontal or vertical axes, remove any text rotations greater than +/- 10 degrees, to improve recognition results.</p> 
                  <p class="datatypelist"><strong>Data Types: </strong><code itemprop="datatype">single</code> | <code itemprop="datatype">double</code> | <code itemprop="datatype">int16</code> | <code itemprop="datatype">uint8</code> | <code itemprop="datatype">uint16</code> | <code itemprop="datatype">logical</code></p>
                 </div>
                </div>
               </div>
              </div>
             </div>
             <div class="panel-group" role="tablist" aria-multiselectable="true">
              <div class="panel panel-default " id="bt548t1-1-roi">
               <div itemscope itemprop="inputargument" itemtype="http://www.mathworks.com/help/schema/MathWorksDocPage/NamedInputArgument" itemid="input_argument_d119e139880">
                <div role="tab" data-toggle="collapse" id="expand_panel_heading_input_argument_d119e139880" data-target="#expand_panel_body_input_argument_d119e139880" aria-controls="expand_panel_body_input_argument_d119e139880" class="panel-heading add_cursor_pointer" aria-expanded="true">
                 <span id="inputarg_roi" class="'anchor_target'"></span>
                 <h3 class="panel-title" id="input_argument_d119e139880"><code itemprop="name">roi</code> — <span itemprop="purpose">Region of interest</span><br> <span class="add_font_color_general remove_bold"><span class="example_desc"><span itemprop="inputvalue"><em class="varname">M</em>-by-4 element matrix</span></span></span></h3>
                 <span class="icon-arrow-open-down icon_24"></span>
                 <span class="icon-arrow-open-right icon_24"></span>
                </div>
                <div role="tabpanel" id="expand_panel_body_input_argument_d119e139880" aria-labelledby="expand_panel_heading_input_argument_d119e139880" class="panel-collapse collapse in">
                 <div class="panel-body"> 
                  <p>One or more rectangular regions of interest, specified as an <em class="varname">M</em>-by-4 element matrix. Each row, <em class="varname">M</em>, specifies a region of interest within the input image, as a four-element vector, [<em class="varname">x</em> <em class="varname">y</em> <em class="varname">width</em> <em class="varname">height</em>]. The vector specifies the upper-left corner location, [<em class="varname">x</em> <em class="varname">y</em>], and the size of a rectangular region of interest, [<em class="varname">width</em> <em class="varname">height</em>], in pixels. Each rectangle must be fully contained within the input image, <a href="#inputarg_I" class="intrnllnk"><code class="argument">I</code></a>. Before the recognition process, the function uses the Otsu’s thresholding to convert truecolor and grayscale input regions of interest to binary regions. The function returns text recognized in the rectangular regions as an array of objects.</p> 
                 </div>

You can specify several name and value pair arguments in any order as <code class="literal">Name1,Value1,...,NameN,ValueN</code>.</p>

                   <div role="tab" data-toggle="collapse" id="expand_panel_heading_input_argument_namevalue_d119e140464" data-target="#expand_panel_body_input_argument_namevalue_d119e140464" aria-controls="expand_panel_body_input_argument_namevalue_d119e140464" class="panel-heading add_cursor_pointer" aria-expanded="true">
                    <span id="inputarg_CharacterSet" class="'anchor_target'"></span>
                    <h3 class="panel-title" id="input_argument_namevalue_d119e140464"><code itemprop="name">'CharacterSet'</code> — <span itemprop="purpose">Character subset</span><br> <span class="add_font_color_general remove_bold"><span class="example_desc"><span itemprop="inputvalue defaultvalue"><code class="literal">''</code> all characters</span> (default) | <span itemprop="inputvalue">character vector</span></span></span></h3>
                    <span class="icon-arrow-open-down icon_24"></span>
                    <span class="icon-arrow-open-right icon_24"></span>
                   </div>
                   <div role="tabpanel" id="expand_panel_body_input_argument_namevalue_d119e140464" aria-labelledby="expand_panel_heading_input_argument_namevalue_d119e140464" class="panel-collapse collapse in">
                    <div class="panel-body"> 
                     <p>Character subset, specified as the comma-separated pair consisting of '<code class="argument">CharacterSet</code>' and a character vector. By default, <code class="argument">CharacterSet</code> is set to the empty character vector, <code class="literal">''</code>. The empty vector sets the function to search for all characters in the language specified by the <a href="#inputarg_Language" class="intrnllnk"><code class="argument">Language</code></a> property. You can set this property to a smaller set of known characters to constrain the classification process.</p> 
                     <p>The <code class="function">ocr</code> function selects the best match from the <code class="argument">CharacterSet</code>. Using deducible knowledge about the characters in the input image helps to improve text recognition accuracy. For example, if you set <code class="argument">CharacterSet</code> to all numeric digits, <code class="literal">'0123456789'</code>, the function attempts to match each character to only digits. In this case, a non-digit character can incorrectly get recognized as a digit.</p> 
                    </div>
                   </div>
                  </div>
                 </div>
                </div>
               </div>
              </div>
             </div>
            </div>
           </div>
          </div>
          <div class="ref_sect">
           <h2 id="bt548t1-4">Output Arguments</h2>
           <div class="expandableContent">
            <p class="switch "><a href="javascript:void(0);" class="expandAllLink" data-allexpanded="true">collapse all</a></p>
            <div class="panel-group" id="bt548t1-4_panel-group" role="tablist" aria-multiselectable="true">
             <div class="panel-group" role="tablist" aria-multiselectable="true">
              <div class="panel panel-default " id="bt548t1-1-txt">
               <div role="tab" data-toggle="collapse" id="expand_panel_heading_output_argument_d119e140522" data-target="#expand_panel_body_output_argument_d119e140522" aria-controls="expand_panel_body_output_argument_d119e140522" class="panel-heading add_cursor_pointer" aria-expanded="true">
                <span id="outputarg_txt" class="'anchor_target'"></span>
                <h3 class="panel-title" id="output_argument_d119e140522"><code class="name">txt</code> — Recognized text and metrics<br> <span class="add_font_color_general remove_bold"><span class="example_desc"><code class="literal">ocrText</code> object</span></span></h3>
                <span class="icon-arrow-open-down icon_24"></span>
                <span class="icon-arrow-open-right icon_24"></span>
               </div>
               <div role="tabpanel" id="expand_panel_body_output_argument_d119e140522" aria-labelledby="expand_panel_heading_output_argument_d119e140522" class="panel-collapse collapse in">
                <div class="panel-body"> 
                 <p>Recognized text and metrics, returned as an <code class="class">ocrText</code> object. The object contains the recognized text, the location of the recognized text within the input image, and the metrics indicating the confidence of the results. The confidence values range is [0 1] and represents a percent probability. When you specify an <em class="varname">M</em>-by-4 <a href="#inputarg_roi" class="intrnllnk"><code class="argument">roi</code></a>, the function returns <code class="class">ocrText</code> as an <em class="varname">M</em>-by-1 array of <code class="class">ocrText</code> objects.</p> 
                 <p>If your <a href="ocr.html"><code class="function">ocr</code></a> results are not what you expect, try one or more of the following options:</p> 
                 <div class="itemizedlist">
                  <ul>
                   <li><p>Increase the image 2-to-4 times the original size.</p></li>
                   <li><p>If the characters in the image are too close together or their edges are touching, use morphology to thin out the characters. Using morphology to thin out the characters separates the characters.</p></li>
                   <li><p>Use binarization to check for non-uniform lighting issues. Use the <a href="../../images/ref/graythresh.html"><code class="function">graythresh</code></a> and <a href="../../images/ref/imbinarize.html"><code class="function">imbinarize</code></a> functions to binarize the image. If the characters are not visible in the results of the binarization, it indicates a potential non-uniform lighting issue. Try top hat, using the <a href="../../images/ref/imtophat.html"><code class="function">imtophat</code></a> function, or other techniques that deal with removing non-uniform illumination.</p></li>
                   <li><p>Use the region of interest <code class="literal">roi</code> option to isolate the text. Specify the <code class="literal">roi</code> manually or use text detection.</p></li>
                   <li><p>If your image looks like a natural scene containing words, like a street scene, rather than a scanned document, try using an ROI input. Also, you can set the <code class="literal">TextLayout</code> property to <code class="literal">'Block'</code> or <code class="literal">'Word'</code>.</p></li>
                  </ul>
                 </div>
                 <p> </p> 
                </div>
               </div>
              </div>
