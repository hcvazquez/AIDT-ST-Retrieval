	];
	
	
	function carrier_onChange(carrier) {
		var form = carrier.form;		
		var disable = false;
		if (carrier.value == "MUIRS") {
			// disable fields 2-5
			disable = true;
		}		
		for (var i = 2; i <= 5; i++) {
			var field = form["trackingNumber" + i];
			field.disabled = disable;			
			if (disable) {
				field.value = "";
			}
		}
	}
	
	
	function trackPackages() {
		var form = document.packageTrackingForm;
				
		var valid = validatePackageTrackingForm(form);
		
		if (valid) {
			var carrier = form.carrier;
			var trackingNumbers = [escape(form.trackingNumber1.value),
								   escape(form.trackingNumber2.value),
								   escape(form.trackingNumber3.value),
								   escape(form.trackingNumber4.value),
								   escape(form.trackingNumber5.value)];
			openCarrierURL(carrier.selectedIndex - 1, trackingNumbers);
		}
	}
	
	function validatePackageTrackingForm(form) {
	
		var valid = true;
		var carrier = form.carrier;
		
		if (carrier.selectedIndex == 0) {
			valid = false;
			alert("Please select a carrier.");
		} else {
			form.trackingNumber1.value = trim(form.trackingNumber1.value);
			form.trackingNumber2.value = trim(form.trackingNumber2.value);
			form.trackingNumber3.value = trim(form.trackingNumber3.value);
			form.trackingNumber4.value = trim(form.trackingNumber4.value);
			form.trackingNumber5.value = trim(form.trackingNumber5.value);
			
			if (form.trackingNumber1.value.length == 0 &&
				form.trackingNumber2.value.length == 0 &&
				form.trackingNumber3.value.length == 0 &&
				form.trackingNumber4.value.length == 0 &&
				form.trackingNumber5.value.length == 0) {
				
				valid = false;
				alert("At least one Tracking Number must be entered.");
			}
		}
		
		return valid;
	}
	
	
	function openCarrierURL(carrierIndex, trackingNumbers) {
		var url = arrCarrierURLs[carrierIndex];
		url = replaceMsgParams(url, trackingNumbers);
		// open a popup window with the URL and no options
		window.open(url, "package_tracking");
	}
