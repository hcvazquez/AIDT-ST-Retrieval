    // Mute video
    var tag = document.createElement('script');
    tag.src = 'https://www.youtube.com/iframe_api';
    var firstScriptTag = document.getElementsByTagName('script')[0];
    firstScriptTag.parentNode.insertBefore(tag, firstScriptTag);

    // Create YouTube player(s) after the API code downloads.
    var player;
    function videoLoadedCb($, ytPlayer){
        $('#mute-btn').click(function(e){
            e.preventDefault();
            if ($(this).hasClass('unmuted')){
                ytPlayer.mute();
                $(this).removeClass('unmuted');
            } else {
                ytPlayer.unMute();
                $(this).addClass('unmuted');
            }
        });
    };

    function getYouTubeVideoId() {
        var id = '';
        var url = $j(".video-section iframe").attr("src");

        if(url) {
            url = url.replace(/(>|<)/gi,'').split(/(vi\/|v=|\/v\/|youtu\.be\/|\/embed\/)/);
            if(url[2] !== undefined) {
                id = url[2].split(/[^0-9a-z_\-]/i);
                id = id[0];
            }
            else {
                id = url;
            }
        }

        return id;
    };

    function onYouTubeIframeAPIReady() {
        player = new YT.Player('yt-player', {
            videoId: getYouTubeVideoId(),
            events: {
                'onReady': onPlayerReady,
                'onStateChange': onPlayerStateChange
            }
        });

        function onPlayerReady(event) {
            event.target.mute();
            event.target.playVideo();
            videoLoadedCb(jQuery, event.target);
        }

        var done = false;
        function onPlayerStateChange(event) {
            if (event.data == YT.PlayerState.PLAYING && !done) {
                done = true;
            }
        }

        function stopVideo() {
            player.stopVideo();
        }
    };
