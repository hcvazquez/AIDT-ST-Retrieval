//Pull all links from the body for easy retrieval
private ArrayList pullLinks(String text) {
ArrayList links = new ArrayList();

String regex = "\\(?\\b(http://|www[.])[-A-Za-z0-9+&amp;amp;@#/%?=~_()|!:,.;]*[-A-Za-z0-9+&amp;amp;@#/%=~_()|]";
Pattern p = Pattern.compile(regex);
Matcher m = p.matcher(text);
while(m.find()) {
String urlStr = m.group();
if (urlStr.startsWith("(") &amp;amp;&amp;amp; urlStr.endsWith(")"))
{
urlStr = urlStr.substring(1, urlStr.length() - 1);
}
links.add(urlStr);
}
return links;

               <p>import java.util.regex.*;</p> 
               <p>public class Replacement {<br> public static void main(String[] args) throws Exception {</p> 
               <p> // Create a pattern to match cat<br> Pattern p = Pattern.compile(“\\((dot\\))|\\[dot\\]|\\{dot\\}”);</p> 
               <p> // Create a matcher with an input string<br> Matcher m = p.matcher(“www(dot)example(dot)com www[dot]example[dot]com www{dot}example{dot}com”);</p> 
               <p> // Loop through and create a new String with the replacements<br> boolean result = m.find();<br> StringBuffer sb = new StringBuffer();<br> while(result) {<br> m.appendReplacement(sb, “.”);<br> result = m.find();<br> }</p> 
               <p> // Add the last segment of input to the new String<br> m.appendTail(sb);<br> System.out.println(sb.toString());<br> }<br> }</p>

         <p>import java.util.regex.*;</p> 
         <p>public class Replacement {<br> public static void main(String[] args) throws Exception {</p> 
         <p> // Create a pattern to match cat<br> Pattern p = Pattern.compile(“\\((dot\\))|\\[dot\\]|\\{dot\\}”);</p> 
         <p> // Create a matcher with an input string<br> Matcher m = p.matcher(“www(dot)example(dot)com www[dot]example[dot]com www{dot}example{dot}com”);</p> 
         <p> // Loop through and create a new String with the replacements<br> boolean result = m.find();<br> StringBuffer sb = new StringBuffer();<br> while(result) {<br> m.appendReplacement(sb, “.”);<br> result = m.find();<br> }</p> 
         <p> // Add the last segment of input to the new String<br> m.appendTail(sb);<br> System.out.println(sb.toString());<br> }<br> }</p>
