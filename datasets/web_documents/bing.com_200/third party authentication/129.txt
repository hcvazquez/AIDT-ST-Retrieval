             <br>

             <br> 
             <br> web.xml
             <br> &lt;context-param&gt;
             <br> &lt;param-name&gt;J2EE_ROLE_NAME
             <wbr>&lt;/param-na
             <wbr>me&gt;
             <br> &lt;param-value&gt;WEBTA_J2EE_US

             <wbr>value&gt;
             <br> &lt;/context-param&gt;
