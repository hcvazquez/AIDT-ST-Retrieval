   <div id="how-it-works" class="section"> 
    <div class="container"> 
     <div class="content"> 
      <div class="text-center"> 
       <h2 class="title">How It Works</h2> 
      </div> 
      <div class="row"> 
       <div class="col-md-12 col-sm-12 col-xs-12"> 
        <p class="description"> In a matter of minutes, you can have an all new, beautiful translator profile, as well as access to our alignment tool and client matching services. </p> 
       </div> 
       <div class="col-md-2 col-sm-2 col-xs-12"></div> 
      </div> 
      <div class="row"> 
       <div class="col-md-6 col-sm-6 col-xs-12"> 
        <div class="how-it-works-step"> 
         <div class="icon-wrapper"> 
          <span class="circle"><i class="fa fa-user-plus"></i></span> 
         </div> 
         <h3>1. Create an account</h3> 
         <p> <a href="/translators/sign_up">Sign up here</a>, and follow the instructions to set up your language pairs and fields of expertise. </p> 
        </div> 
       </div> 
       <div class="col-md-6 col-sm-6 col-xs-12"> 
        <div class="how-it-works-step"> 
         <div class="icon-wrapper"> 
          <span class="circle"><i class="fa fa-check-square-o"></i></span> 
         </div> 
         <h3>2. Opt-in to Nak?do</h3> 
         <p>Keep an eye out for a check-box to be listed in Nak?do. If you aren't listed, you won't be shown in Nak?do search results.</p> 
        </div> 
       </div> 
       <div class="col-md-6 col-sm-6 col-xs-12"> 
        <div class="how-it-works-step"> 
         <div class="icon-wrapper"> 
          <span class="circle"><i class="fa fa-upload"></i></span> 
         </div> 
         <h3>3. Load or create TMs</h3> 
         <p> Load a translation memory (TM) file into your private TM-Town account. TMs are how we index you by your expertise in Nak?do. Keep in mind, you can create TMs with our alignment tool, and we also offer a redactor to exclude confidential terms. If you don't have work you can load directly to your account, start with our <a href="/sample-translations">Sample Translations</a> or check out <a href="/desktop-application">Deshi</a>, our desktop application. </p> 
        </div> 
       </div> 
       <div class="col-md-6 col-sm-6 col-xs-12"> 
        <div class="how-it-works-step"> 
         <div class="icon-wrapper"> 
          <span class="circle"><i class="fa fa-refresh"></i></span> 
         </div> 
         <h3>4. (Optional) Sync with a CAT tool</h3> 
         <p>You can search your terms and segments from our currently supported CAT tools, CafeTran or SDL Trados 2015, or directly from your web browser.</p> 
        </div>

  (function($){
    $(function(){

      var navbarHeight = 82;

      // Slide in Functionality
      $(window).scroll(function() {
        var top = $(window).scrollTop();
        $(".slide-in").each(function () {
          var thisTop = $(this).offset().top;
          var height = $(this).height();
          if ((top > (thisTop - (height * 1.5))) && !$(this).hasClass("slid")) {
            $(this).addClass("slid");
          }
        });
        $('body').scrollspy({ offset: navbarHeight+10, target: '#navbar' });
      });

      // Navbar Affix
      $('#navbar').affix({
        offset: {
          top: function () {
            return (this.top = $(window).height() - navbarHeight)
          }
        }
      })

      // Set Home section height
      $("#home").css('height', $(window).height() - 52 + 'px')

      // Parallax Scripts
      function updateParallax() {
        if ($(window).width() > 768) {
          $(".parallax").each(function () {
            var bottom = $(this).offset().top + $(this).height();
            var top = $(this).offset().top;
            var windowHeight = $(window).height();
            var scrollTop = $(window).scrollTop() + windowHeight;
            var fromTop = 82;
            var isHome = true;
            if (top === 0) {
              fromTop = $(window).scrollTop() - top;
              isHome = true;
            }
            else {
              fromTop = $(window).scrollTop() - top + windowHeight;
              isHome = false;
            }
            if ((bottom > $(window).scrollTop()) && (top < scrollTop)) {
              var parallax = -1 * (fromTop / 3);
              var revParallax = parallax;
              var percent = 1 - 1.3 *($(window).scrollTop() / $(window).height());
              if (isHome) {
                revParallax += navbarHeight;
                $("#home .logo").css('marginTop', parallax + "px");
                $("#home .logo, #home .welcome, #home .call-to-action, #home .macbook-preview, #home .parralax, #home .community-square, .timer").css('opacity', percent);
                $('#home .community-square').css('top', revParallax - 40 + "px");
              }
              $(this).children("img").first().css('bottom', revParallax + "px");
            }
          });
        }
      }
      updateParallax();

      $(window).scroll(function() {
        updateParallax();
      });

      // Smooth Scrolling
      $("a.scroll").click(function(e) {
        e.preventDefault();
        var offset = $(this.hash).offset().top - (navbarHeight/2);
        $('html, body').animate({ scrollTop: offset }, 600);
      });

      // Accordion Active Toggling
      $("a[data-toggle='collapse']").click(function() {
        if ($(this).parent().parent(".panel-heading").hasClass("active")) {
          $(this).parent().parent().parent().parent().find('.panel-heading').removeClass("active");
        }
        else {
          $(this).parent().parent().parent().parent().find('.panel-heading').removeClass("active");
          $(this).parent().parent(".panel-heading").addClass("active");
        }
      });

    }); // end of document ready
  })(jQuery); // end of jQuery name space
</script>

    function animateActivity(cssClass, container) {
      // Uploads activity animation
      var arr = [];
      $(cssClass).each(function() {
        arr.push(this.id);
      });
      var activities = arr;

      $("#" + activities[0]).show();
      $("#" + activities[1]).show();
      $("#" + activities[2]).show();
      $("#" + activities[3]).show();

      var counter = 3;

      function startTimer() {
        timer = setInterval(function() {
          $("#" + activities[counter]).hide().prependTo(container).slideDown(1500);

          if (counter < activities.length) {
            counter += 1;
            randomTime = (Math.floor((Math.random() * 8000) + 4500))
          } else {
            counter = 0;
          }
        }, randomTime);
      }

      startTimer();
    }

    animateActivity('.column1', '#column1Activities');

    animateActivity('.column3', '#column3Activities');
