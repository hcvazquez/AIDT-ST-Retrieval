	}
}


function right_click(obj){
	var intro = document.getElementById('intro').innerHTML;
	var attention = obj.innerHTML;
	if(intro == '' && attention == 'Next'){
		return false;
	}
	id=parseInt(intro)+1;
	$(".current").removeClass("current");
	$("#"+id).parent().parent().addClass("current");
	tab=$("#"+id).attr("src_big");
	intro_p=$("#"+id).attr("title");
	$("#showimg").attr("src",tab);
	$(".intro").text(intro_p);
	address();
};

function left_click(obj){
	var intro = document.getElementById('intro').innerHTML;
	var attention = obj.innerHTML;
	if(intro == '' && attention == 'Next'){
		return false;
	}
	if(intro == '1' && attention == 'Previous'){
		return false;
	}
	id=intro;
	if(attention == 'Next'){
		document.getElementById("jiazaizhong").style.display="block";
		id=parseInt(intro)+1;
		$(".current").removeClass("current");
		$("#"+id).parent().parent().addClass("current");

	}else{
		document.getElementById("jiazaizhong").style.display="block";
		id=id-1;
		$(".current").removeClass("current");
		$("#"+id).parent().parent().addClass("current");
	}
	$("#p_n").html(id);
	tab=$("#"+id).attr("src_big");
	intro_p=$("#"+id).attr("title");
	$("#showimg").attr("src",tab);
	$(".intro").text(intro_p);
	address();
};


function bottom_click(){
	top_px=parseInt($("#mycarousel ul").css("top"));
	now_px=((top_px/116)-1)*116;
	if(now_px>(-((len-4)*116))){
		$("#mycarousel ul").animate({top:now_px},500);
	}else{
		$("#mycarousel ul").animate({top:0},2000)
	}
}

function top_click(){
	top_px=parseInt($("#mycarousel ul").css("top"));
	now_px=((top_px/110)+1)*116;
	if(now_px>0){
		now_px=-((len-4)*116);
		$("#mycarousel ul").animate({top:now_px},2000);
	}else{
		$("#mycarousel ul").animate({top:now_px},500)
	}
}

function a_link(){
	id=$(this).children("img").attr("id")-0;
	top_px=parseInt($("#mycarousel ul").css("top"));
	$(".current").removeClass("current");
	$(this).parents("li").addClass("current");
	tab=$("#"+id).attr("src_big");
	intro_p=$("#"+id).attr("title");
	$("#showimg").attr("src",tab);
	$(".intro").text(intro_p);
	if((id+(top_px/116))>4){
		bottom_click();
	}
}

function slide_load(id){

	})
}
