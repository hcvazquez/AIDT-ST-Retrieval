function prebid_bidder_settings_configuration()
{
  return {
    standard: {
      adserverTargeting: [{
        key: "hb_bidder",
        val: function(bid) {
          return bid.bidderCode
        }
      }, {
        key: "hb_adid",
        val: function(bid) {
          return bid.adId
        }
      }, {
        key: "hb_pb",
        val: function(bid) {
          var hb_pb = (Math.min(PREBID_MAXIMUM, bid.cpm)).toFixed(2);
          if (debug_hb >= 2)
          {
            var too_late = (bid.timeToResponse >= PREBID_TIMEOUT) ? " TOO LATE!" : ""; 
            console.info(bid.adUnitCode+" "+bid.bidder+" "+hb_pb+" time: "+bid.timeToRespond+too_late);
          }
          return hb_pb;
        }
      }]
    },
    aol: {
      bidCpmAdjustment: function(bidCpm, bid) {
        cpm2 = (Math.ceil(bidCpm * PREBID_CENTS) / PREBID_CENTS).toFixed(2);
        return parseFloat(cpm2);
      },
    },
    brealtime: {
      bidCpmAdjustment : function(bidCpm, bid){
        var cpm2 = 0.85*bidCpm;
        cpm2 = (Math.ceil(cpm2 * PREBID_CENTS) / PREBID_CENTS).toFixed(2);
        return parseFloat(cpm2);
      }
    },
    conversant: {
      bidCpmAdjustment: function(bidCpm, bid) {
