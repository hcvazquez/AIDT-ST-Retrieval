   <div class="sect1"> 
    <h2 id="installation"><a class="link" href="#installation">1. Installation</a></h2> 
    <div class="sectionbody"> 
     <div class="paragraph"> 
      <p>Download the <a href="http://jexcelapi.sourceforge.net/">Java Excel library</a> from the webpage</p> 
     </div> 
     <div class="paragraph"> 
      <p>To use this library in your Java program add the lib jxl.jar to your classpath in your project. See <a href="http://www.vogella.com/tutorials/Eclipse/article.html#classpath">Changing classpath in Eclipse</a>.</p> 
     </div> 
    </div> 
   </div> 
   <div class="sect1"> 
    <h2 id="createexcel"><a class="link" href="#createexcel">2. Create an Excel Spreadsheet</a></h2> 
    <div class="sectionbody"> 
     <div class="paragraph"> 
      <p>Create a new Java project called <em>de.vogella.java.excel</em>. Create the <code>de.vogella.java.excel.writer</code> package and the following class.</p> 
     </div> 
     <div class="listingblock"> 
      <div class="content"> 
       <pre class="CodeRay highlight"><code data-lang="java"><span class="keyword">package</span> <span class="namespace">writer</span>;

<span class="keyword">import</span> <span class="include">java.io.File</span>;
<span class="keyword">import</span> <span class="include">java.io.IOException</span>;
<span class="keyword">import</span> <span class="include">java.util.Locale</span>;

<span class="keyword">import</span> <span class="include">jxl.CellView</span>;
<span class="keyword">import</span> <span class="include">jxl.Workbook</span>;
<span class="keyword">import</span> <span class="include">jxl.WorkbookSettings</span>;
<span class="keyword">import</span> <span class="include">jxl.format.UnderlineStyle</span>;
<span class="keyword">import</span> <span class="include">jxl.write.Formula</span>;
<span class="keyword">import</span> <span class="include">jxl.write.Label</span>;
<span class="keyword">import</span> <span class="include">jxl.write.Number</span>;
<span class="keyword">import</span> <span class="include">jxl.write.WritableCellFormat</span>;
<span class="keyword">import</span> <span class="include">jxl.write.WritableFont</span>;
<span class="keyword">import</span> <span class="include">jxl.write.WritableSheet</span>;
<span class="keyword">import</span> <span class="include">jxl.write.WritableWorkbook</span>;
<span class="keyword">import</span> <span class="include">jxl.write.WriteException</span>;
<span class="keyword">import</span> <span class="include">jxl.write.biff.RowsExceededException</span>;


<span class="directive">public</span> <span class="type">class</span> <span class="class">WriteExcel</span> {

    <span class="directive">private</span> WritableCellFormat timesBoldUnderline;
    <span class="directive">private</span> WritableCellFormat times;
    <span class="directive">private</span> <span class="predefined-type">String</span> inputFile;

    <span class="local-variable">this</span>.inputFile = inputFile;
    }

    <span class="directive">public</span> <span class="type">void</span> write() <span class="directive">throws</span> <span class="exception">IOException</span>, WriteException {
        <span class="predefined-type">File</span> file = <span class="keyword">new</span> <span class="predefined-type">File</span>(inputFile);
        WorkbookSettings wbSettings = <span class="keyword">new</span> WorkbookSettings();

        wbSettings.setLocale(<span class="keyword">new</span> <span class="predefined-type">Locale</span>(<span class="string"><span class="delimiter">"</span><span class="content">en</span><span class="delimiter">"</span></span>, <span class="string"><span class="delimiter">"</span><span class="content">EN</span><span class="delimiter">"</span></span>));

        WritableWorkbook workbook = Workbook.createWorkbook(file, wbSettings);
        workbook.createSheet(<span class="string"><span class="delimiter">"</span><span class="content">Report</span><span class="delimiter">"</span></span>, <span class="integer">0</span>);

        <span class="comment">// Define the cell format</span>

        <span class="comment">// Lets automatically wrap the cells</span>
        times.setWrap(<span class="predefined-constant">true</span>);

        <span class="comment">// create create a bold font with unterlines</span>
        WritableFont times10ptBoldUnderline = <span class="keyword">new</span> WritableFont(
                WritableFont.TIMES, <span class="integer">10</span>, WritableFont.BOLD, <span class="predefined-constant">false</span>,
                UnderlineStyle.SINGLE);
        timesBoldUnderline = <span class="keyword">new</span> WritableCellFormat(times10ptBoldUnderline);
        <span class="comment">// Lets automatically wrap the cells</span>
        timesBoldUnderline.setWrap(<span class="predefined-constant">true</span>);

        CellView cv = <span class="keyword">new</span> CellView();
        cv.setFormat(times);
        cv.setFormat(timesBoldUnderline);
        cv.setAutosize(<span class="predefined-constant">true</span>);

        <span class="comment">// Write a few headers</span>
        addCaption(sheet, <span class="integer">0</span>, <span class="integer">0</span>, <span class="string"><span class="delimiter">"</span><span class="content">Header 1</span><span class="delimiter">"</span></span>);
        addCaption(sheet, <span class="integer">1</span>, <span class="integer">0</span>, <span class="string"><span class="delimiter">"</span><span class="content">This is another header</span><span class="delimiter">"</span></span>);


    }

    <span class="directive">private</span> <span class="type">void</span> createContent(WritableSheet sheet) <span class="directive">throws</span> WriteException,
            RowsExceededException {
        <span class="comment">// Write a few number</span>
        <span class="keyword">for</span> (<span class="type">int</span> i = <span class="integer">1</span>; i &lt; <span class="integer">10</span>; i++) {

            addNumber(sheet, <span class="integer">0</span>, i, i + <span class="integer">10</span>);
            <span class="comment">// Second column</span>
            addNumber(sheet, <span class="integer">1</span>, i, i * i);
        }
        <span class="comment">// Lets calculate the sum of it</span>
        <span class="predefined-type">StringBuffer</span> buf = <span class="keyword">new</span> <span class="predefined-type">StringBuffer</span>();
        buf.append(<span class="string"><span class="delimiter">"</span><span class="content">SUM(A2:A10)</span><span class="delimiter">"</span></span>);
        Formula f = <span class="keyword">new</span> Formula(<span class="integer">0</span>, <span class="integer">10</span>, buf.toString());
        sheet.addCell(f);
        buf = <span class="keyword">new</span> <span class="predefined-type">StringBuffer</span>();
        buf.append(<span class="string"><span class="delimiter">"</span><span class="content">SUM(B2:B10)</span><span class="delimiter">"</span></span>);
        f = <span class="keyword">new</span> Formula(<span class="integer">1</span>, <span class="integer">10</span>, buf.toString());
        sheet.addCell(f);

        <span class="comment">// now a bit of text</span>
        <span class="keyword">for</span> (<span class="type">int</span> i = <span class="integer">12</span>; i &lt; <span class="integer">20</span>; i++) {
            <span class="comment">// First column</span>
            addLabel(sheet, <span class="integer">0</span>, i, <span class="string"><span class="delimiter">"</span><span class="content">Boring text </span><span class="delimiter">"</span></span> + i);
            <span class="comment">// Second column</span>
            addLabel(sheet, <span class="integer">1</span>, i, <span class="string"><span class="delimiter">"</span><span class="content">Another text</span><span class="delimiter">"</span></span>);
        }
    }

    <span class="directive">private</span> <span class="type">void</span> addCaption(WritableSheet sheet, <span class="type">int</span> column, <span class="type">int</span> row, <span class="predefined-type">String</span> s)
            <span class="directive">throws</span> RowsExceededException, WriteException {

        sheet.addCell(label);
    }

    <span class="directive">private</span> <span class="type">void</span> addNumber(WritableSheet sheet, <span class="type">int</span> column, <span class="type">int</span> row,
            <span class="predefined-type">Integer</span> integer) <span class="directive">throws</span> WriteException, RowsExceededException {
        <span class="predefined-type">Number</span> number;
        number = <span class="keyword">new</span> <span class="predefined-type">Number</span>(column, row, integer, times);
        sheet.addCell(number);
    }

    <span class="directive">private</span> <span class="type">void</span> addLabel(WritableSheet sheet, <span class="type">int</span> column, <span class="type">int</span> row, <span class="predefined-type">String</span> s)
            <span class="directive">throws</span> WriteException, RowsExceededException {
        <span class="predefined-type">Label</span> label;
        label = <span class="keyword">new</span> <span class="predefined-type">Label</span>(column, row, s, times);
        sheet.addCell(label);
    }
