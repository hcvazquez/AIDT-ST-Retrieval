                    $this.addClass('active');

        // search action helper for new Cortana Intelligence search bar
        require(['jquery'], function ($) {
            var $cortanaNavbar = $('.cortana-top-nav');
            if ($cortanaNavbar.length) {
                var $waffleBtn = $cortanaNavbar.find('.waffle-btn');
                $cortanaNavbar.find('.search-bar-toggle').click(function () {
                    $waffleBtn.removeClass('open');
                    $cortanaNavbar.toggleClass('in-search');
                });

                $waffleBtn.find('a').not('[data-toggle]').click(function () {
                    $waffleBtn.removeClass('open');
                })

                $cortanaNavbar.find('.search-bar-close').click(function () {
                    $cortanaNavbar.removeClass('in-search');
                });

                var $searchInput = $cortanaNavbar.find('.top-nav-search-input');
                $cortanaNavbar.find('.top-nav-search-submit').click(function () {
                    searchText = $searchInput.val().trim();
                    window.location.href = "/browse" + (searchText ? ("?s=" + encodeURIComponent(searchText)) : "");
                });

            }

            var $navigationNotfication = $('.navbar-notification');
            if ($navigationNotfication.length) {
                $navigationNotfication.find('.navbar-notification-close-button').click(function (e) {
                    e.preventDefault();
                    $navigationNotfication.slideUp(400);

                    var currentDate = new Date();
                    currentDate.setYear(currentDate.getFullYear() + 1);
                    document.cookie = "hide-nav-notification=true; path=/; expires=" + currentDate.toUTCString();
                });
            }
        });
