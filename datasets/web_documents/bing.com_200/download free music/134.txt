	function uncheckAll(form) {
		$('#form_domains_' + form + ' input').removeClass('selected');
		$('#form_domains_' + form + ' label').removeClass('selected');
	}
	
	function check(form) {
		return function() {
			uncheckAll(form);
			
			var inputs = $('#form_domains_' + form + ' input');
			var no;
			for (var i = 0; i < inputs.length; i++) {
				if (inputs[i].checked) {
					no = $(inputs[i]).attr('id').substr($(inputs[i]).attr('id').lastIndexOf('_') + 1);
					$(inputs[i]).parent().find('input#domain_radio_'+form+'_' + no).addClass('selected');
					$(inputs[i]).parent().find('label#l_domain_radio_'+form+'_' + no).addClass('selected');
					break;
				}
			}
		};
	}
