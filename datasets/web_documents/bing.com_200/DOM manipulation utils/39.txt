import org.w3c.dom.Document;
import org.w3c.dom.NamedNodeMap;
import org.w3c.dom.Node;
import org.w3c.dom.NodeList;
import org.xml.sax.SAXException;
import sh.owl.bot.objects.PlaylistElement;

import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;
import javax.xml.parsers.ParserConfigurationException;
import javax.xml.xpath.*;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

public class PlaylistUtils {
    // singleton
    private static PlaylistUtils instance;

    // something like a xml document ... and apparently that needs a factory
    private DocumentBuilderFactory documentBuilderFactory;
    private DocumentBuilder documentBuilder;

    // xpath is a really nice way to find stuff in xml: https://www.w3schools.com/xml/xpath_intro.asp
    private XPathFactory xPathFactory;
    private  XPath xPath;

    // expressions are those strings you write to find specific elements in the xml
    // check https://www.w3schools.com/xml/xpath_syntax.asp
    private XPathExpression representationExpression;
    private XPathExpression baseUrlExpression;

    private PlaylistUtils() {
        // create factory
        documentBuilderFactory = DocumentBuilderFactory.newInstance();
        documentBuilderFactory.setNamespaceAware(true); // dunno
        try {
            // create document
            documentBuilder = documentBuilderFactory.newDocumentBuilder();

            // create xpath factory
            xPathFactory = XPathFactory.newInstance();
            // create xpath
            xPath = xPathFactory.newXPath();

            // theres a weird thing in xml called namespaces, basically you can't find stuff if those aren't configured, so we'll bypass that by looking at it directly with local-name
            // "//*" find any element with name "Representation"
            representationExpression = xPath.compile("//*[local-name()='Representation']");
            // "./*" find any name inside the current element with name "BaseURL"
            baseUrlExpression = xPath.compile("./*[local-name()='BaseURL']");
        } catch (ParserConfigurationException | XPathExpressionException e) {
            System.err.println("Something happened during playlist utils init");
            e.printStackTrace();
        }
    }

    // singleton magic
    public static PlaylistUtils getInstance() {
        if (instance == null)
            instance = new PlaylistUtils();
        return instance;
    }

    public List&lt;PlaylistElement&gt; convertStringToPlaylist(String playlist) throws IOException, SAXException, XPathExpressionException {
        // creates that xml document based on the playlist url ( e.g. https://v.redd.it/dbl99tpugjgz/DASHPlaylist.mpd )
        Document doc = documentBuilder.parse(playlist);

        // finds all "Representation"
        NodeList result = (NodeList) representationExpression.evaluate(doc, XPathConstants.NODESET);

        List&lt;PlaylistElement&gt; elements = new ArrayList&lt;&gt;();
        for (int i = 0; i &lt; result.getLength(); i++) {
            // Gets the current "Representation"
            Node item = result.item(i);
            // Get all attributes (bandwidth, codecs, ...) of the item
            NamedNodeMap attributes = item.getAttributes();

            // Finds base url based on the current item
            String baseUrl = (String) baseUrlExpression.evaluate(item, XPathConstants.STRING);

            elements.add(new PlaylistElement(
                    baseUrl,
                    Long.parseLong(attributes.getNamedItem("bandwidth").getTextContent().trim()), // get text content of attribute "bandwidth" and trim it
                    attributes.getNamedItem("codecs").getTextContent().trim(),
                    attributes.getNamedItem("mimeType").getTextContent().trim()
            ));
        }

        return elements;
    }

    public PlaylistElement getHighestBitrate(List&lt;PlaylistElement&gt; elements) {
        elements.sort((t0, t1) -&gt; (int) (t1.getBandwidth() - t0.getBandwidth())); // sorts based on bandwidth

        return elements.get(0); // gets first one
    }
