   <script type="text/javascript">function displayKwindow(tabIndex) {
    var
      ktabs = $(".NavAlt2Tabs-item"),
      kwindow = $(".k-window"),
      elem = ktabs.eq(tabIndex),
      startY = elem.offset().top + 0.32 * elem.height(),
      startX = elem.offset().left + 0.5 * elem.width()
    ;
    kwindow.hide().eq(tabIndex).css({
      "top":startY,
      "left":startX
    }).fadeIn(350);
}

function onDragStart() {
  $(".NavAlt2Tabs--kendo").addClass("is-drag-started");
}

$(document).ready(function(){
  
  var kwindow = $(".k-window");
  kwindow.hide();

  $(".NavAlt2 a").on("click", function(e){
      e.preventDefault();

      if( ! $(e.target).hasClass("is-active") ) {
        $(".NavAlt2Tabs--kendo").removeClass("is-drag-started");
      }

      var $link = $(this),
          tabIndex = $link.parent().index();

      // if( $link.hasClass("is-active") ) { return false; }

      $(".k-window").hide();
      window.setTimeout(function(){displayKwindow(tabIndex)}, 650);

      function changeHash(newHash) {
        newHash = new RegExp("#").test(newHash) ? newHash : "#" + newHash;
        if (window.history.replaceState) {
          window.history.replaceState("", document.title, newHash);
        } else {
          window.location.replace(newHash);
        }
      }
      changeHash($link.attr("href"));

  });

});

$(window).load(function(){
  var
    knav = $(".NavAlt2"),
    kwindow = $(".k-window"),
    tabIndex = knav.find(".is-active").parent().index(),
    loadingBar = $(".js-loading"),
    lazyImages = $(".lazy-image")
  ;
  kwindow.hide();
  loadingBar.hide();
  window.setTimeout(function(){displayKwindow(tabIndex)}, 650);
  $("body").addClass("is-loaded");
  lazyImages.each(function(){
    var $img = $(this);
    $img.attr("src", $img.data("src"));
  });

});

$(".Nav-title").on("click", function() {
  $("html,body").animate({scrollTop:0},400);
});

var _startNavPos = $(".PageWrapper").offset().top,
    kendokas = $(".DividerSection--kendo");
$(window).on("scroll", function(){
  var
    winTop = $(window).scrollTop(),
    fnav = $(".Nav.is-fixed")
  ;
  if( winTop > _startNavPos ) {fnav.fadeIn(250);}
  else {fnav.fadeOut(250);}

  kendokas.each(function(i, kendoka){
    var $kendoka = $(kendoka),
        winH = $(window).height();
    $kendoka.addClass("is-kendoka-" + (i + 1));
    if( $kendoka.offset().top <= winTop + winH - 0.5 * winH ) {
      $kendoka.addClass("is-into-view");
    } else {
      $kendoka.removeClass("is-into-view");
    }
  });
});

function play() {
    var frames = 0,
        start = new Date();

    playInterval = setInterval(function() {
        step();
        frames++;
    }, 100);
}

function stop() {
    clearInterval(playInterval);
    playInterval = null;
}

function step() {
    addPoint();
    var chart = $("#chart").data("kendoChart");
    if (chart) {
        chart.refresh();
    }
}

function addPoint() {
    var stockData,
        change,
        lastValue;

    // Shift existing categories to the left and add the next date at the end
    lastDate = new Date(lastDate.getTime() + TICKS_PER_DAY);
    categoryList.push((lastDate.getMonth() + 1) + "/" + (lastDate.getDay() + 1));

    if (categoryList.length > POINTS) {
        categoryList.shift();
    }

    for (var i = 0; i < stocks.length; i++) {
        stockData = stocks[i];
        change = (Math.random() > 0.5 ? 1 : - 1) * Math.random() * 10;
        lastValue = stockData[stockData.length - 1] || Math.random() * 10;

        // Add a new pseudo-random data point
        stockData.push(Math.min((i + 1) * 20, Math.max((i + 1) * 10, lastValue + change)));

        // Shift the data points of each series to the left
        if (stockData.length > POINTS) {
            stockData.shift();
        }
    }
}
