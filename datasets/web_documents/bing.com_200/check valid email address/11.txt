                    <div id="CtzXuQ"></div>
                    <span class="rendered_qtext">You can simply use the following user defined function(UDF) to validate the email addresses<br><br><br><br>
                     <div class="qtext_image_wrapper">
                      <canvas class="qtext_image_placeholder landscape qtext_image zoomable_in zoomable_in_feed" width="602.0" height="332.9603803486529"></canvas>
                      <img class="landscape qtext_image zoomable_in zoomable_in_feed lazy_loaded lazy_loading" src="" data-src="https://qph.ec.quoracdn.net/main-qimg-adfab4cfb909063653d1a4655a843c3b.webp" master_src="https://qph.ec.quoracdn.net/main-qimg-eaa2f9c69b1b2f079e3576380eb56262-c" master_w="631" master_h="349">
                     </div><br>
                     <hr class="qtext_hr">
                     <ol>
                      <li>Copy the code below.</li>
                      <li>Open any workbook.</li>
                      <li>Press Alt + F11 to open the Visual Basic Editor (VBE).</li>
                      <li>Press Ctrl + R to show the Project Explorer.</li>
                      <li>Right-click desired file on left (in bold).</li>
                      <li>From the Menu, choose Insert-Module.</li>
                      <li>Paste the code into the right-hand code window.</li>
                      <li>Close the VBE, save the file&nbsp; as VBA Enabled WorkBook (xlsm) for Excel 2010 and newer<br><br></li>
                     </ol><b>How to Use the UDF</b><br>
                     <ol>
                      <li>Make sure you have enabled the macros</li>
                      <li>You can use this in a formula on your worksheet as follows =IsEmailValid("<span class="qlink_container"><a href="mailto:someemail@somedomain.com" rel="noopener nofollow" onclick="return Q.openUrl(this, 5398911);" class="external_link">someemail@somedomain.com</a></span>")</li>
                      <li>If the email address is valid, the formula will return TRUE</li>
                     </ol><br><b>CODE</b>
                     <hr class="qtext_hr"><br><br>
                     <blockquote>
                      <br>Function IsEmailValid(strEmail) 
                      <br>&nbsp;&nbsp;&nbsp; Dim strArray As Variant 
                      <br>&nbsp;&nbsp;&nbsp; Dim strItem As Variant 
                      <br>&nbsp;&nbsp;&nbsp; Dim i As Long, c As String, blnIsItValid As Boolean 
                      <br>&nbsp;&nbsp;&nbsp; blnIsItValid = True 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp; 
                      <br>&nbsp;&nbsp;&nbsp; i = Len(strEmail) - Len(Application.Substitute(strEmail, "@", "")) 
                      <br>&nbsp;&nbsp;&nbsp; If i &lt;&gt; 1 Then IsEmailValid = False: Exit Function 
                      <br>&nbsp;&nbsp;&nbsp; Redim strArray(1 To 2) 
                      <br>&nbsp;&nbsp;&nbsp; strArray(1) = Left(strEmail, InStr(1, strEmail, "@", 1) - 1) 
                      <br>&nbsp;&nbsp;&nbsp; strArray(2) = Application.Substitute(Right(strEmail, Len(strEmail) - Len(strArray(1))), "@", "") 
                      <br>&nbsp;&nbsp;&nbsp; For Each strItem In strArray 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; If Len(strItem) &lt;= 0 Then 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; blnIsItValid = False 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; IsEmailValid = blnIsItValid 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; Exit Function 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; End If 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; For i = 1 To Len(strItem) 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; c = LCase(Mid(strItem, i, 1)) 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; If InStr("abcdefghijklmnopqrstuvwxyz_-.", c) &lt;= 0 And Not IsNumeric(c) Then 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; blnIsItValid = False 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; IsEmailValid = blnIsItValid 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; Exit Function 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; End If 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; Next i 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; If Left(strItem, 1) = "." Or Right(strItem, 1) = "." Then 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; blnIsItValid = False 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; IsEmailValid = blnIsItValid 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; Exit Function 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; End If 
                      <br>&nbsp;&nbsp;&nbsp; Next strItem 
                      <br>&nbsp;&nbsp;&nbsp; If InStr(strArray(2), ".") &lt;= 0 Then 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; blnIsItValid = False 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; IsEmailValid = blnIsItValid 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; Exit Function 
                      <br>&nbsp;&nbsp;&nbsp; End If 
                      <br>&nbsp;&nbsp;&nbsp; i = Len(strArray(2)) - InStrRev(strArray(2), ".") 
                      <br>&nbsp;&nbsp;&nbsp; If i &lt;&gt; 2 And i &lt;&gt; 3 Then 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; blnIsItValid = False 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; IsEmailValid = blnIsItValid 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; Exit Function 
                      <br>&nbsp;&nbsp;&nbsp; End If 
                      <br>&nbsp;&nbsp;&nbsp; If InStr(strEmail, "..") &gt; 0 Then 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; blnIsItValid = False 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; IsEmailValid = blnIsItValid 
                      <br>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; Exit Function 
                      <br>&nbsp;&nbsp;&nbsp; End If 
                      <br>&nbsp;&nbsp;&nbsp; IsEmailValid = blnIsItValid 
                      <br>End Function
                     </blockquote></span>
