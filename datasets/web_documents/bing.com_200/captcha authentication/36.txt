          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;@Html.LabelFor(model&nbsp;=&gt;&nbsp;model.Title)&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;/div&gt;&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;div&nbsp;<span class="keyword">class</span><span>=</span><span class="string">"editor-field"</span><span>&gt;&nbsp;&nbsp;</span></span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;@Html.EditorFor(model&nbsp;=&gt;&nbsp;model.Title)&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;@Html.ValidationMessageFor(model&nbsp;=&gt;&nbsp;model.Title)&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;/div&gt;&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;div&nbsp;<span class="keyword">class</span><span>=</span><span class="string">"editor-label"</span><span>&gt;&nbsp;&nbsp;</span></span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;@Html.LabelFor(model&nbsp;=&gt;&nbsp;model.Comment)&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;/div&gt;&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;div&nbsp;<span class="keyword">class</span><span>=</span><span class="string">"editor-field"</span><span>&gt;&nbsp;&nbsp;</span></span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;@Html.TextAreaFor(model&nbsp;=&gt;&nbsp;model.Comment,5,40,<span class="keyword">null</span><span>)&nbsp;&nbsp;</span></span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;@Html.ValidationMessageFor(model&nbsp;=&gt;&nbsp;model.Comment)&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;/div&gt;&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;@Html.MathCaptcha()&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;@*@Html.Captcha(3)*@&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;br&nbsp;/&gt;&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;p&nbsp;<span class="keyword">class</span><span>=</span><span class="string">"Error"</span><span>&gt;&nbsp;&nbsp;@ViewBag.ErrMessage&nbsp;&lt;/p&gt;&nbsp;&nbsp;</span></span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;p&gt;&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;input&nbsp;type=<span class="string">"submit"</span><span>&nbsp;value=</span><span class="string">"Send"</span><span>&nbsp;/&gt;&nbsp;&nbsp;</span></span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;/p&gt;&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&nbsp;&nbsp;&nbsp;&nbsp;&lt;/fieldset&gt;&nbsp;&nbsp;</span> </li>
          <li><span>}&nbsp;&nbsp;</span></li>
         </ol>
        </div>
        <strong>Note:</strong> Here if you use the @Html.MathCaptcha() helper class then it will generate a mathmatical CAPTCHA. If you use the @Html. Captcha(3) helper class then it will generate a Char CAPTCHA. 3 is the length of the CAPTCHA.
        <br>
        <br>
        <strong>Step 6:</strong> Create the ThankYouPage index like this.
        <br> 
        <div class="dp-highlighter"> 
         <div class="bar"></div> 
         <ol class="dp-c"> 
          <li class="alt"><span><span>@model&nbsp;MathCaptcha.Models.Feedback&nbsp;&nbsp;</span></span> </li>
          <li><span>&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>@{&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;&nbsp;&nbsp;ViewBag.Title&nbsp;=&nbsp;<span class="string">"ThankYouPage"</span><span>;&nbsp;&nbsp;</span></span> </li>
          <li class="alt"><span>}&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&lt;h2&gt;ThankYouPage&lt;/h2&gt;&nbsp;&nbsp;</span> </li>
          <li><span>&nbsp;&nbsp;</span> </li>
          <li class="alt"><span>&lt;b&gt;&nbsp;Thank&nbsp;you&nbsp;sending&nbsp;your&nbsp;valuable&nbsp;feedback&nbsp;to&nbsp;us.&lt;/b&gt;&nbsp;&nbsp;</span></li>
         </ol>
        </div>
        <strong>Points to remember</strong>&nbsp; 
        <ol> 
         <li>Just include the CaptchaMvc libray.<br><br> </li>
         <li>Use the CaptchaMvc.HtmlHelpers namespace.<br><br> </li>
         <li>Use the MathCaptcha namespace for using mathCaptcha.<br><br> </li>
         <li>Use the CaptchaMvc namespace for using CharCaptcha.<br><br> </li>
         <li>For a math CAPTCHA use the @Html.MathCaptcha() helper class.<br><br> </li>
         <li>For a Char CAPTCHA use the @Html.Captcha(3) helper class. </li>
        </ol> 
        <p><strong>Summary<br><br></strong>Here we saw how to use a CAPTCHA in an ASP.NET MVC application. It is very simple and straight forward to implement in MVC applications.</p>
