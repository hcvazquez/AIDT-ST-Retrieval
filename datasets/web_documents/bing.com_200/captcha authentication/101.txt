               <br> &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; var apiConsumer = new ApiConsumer();

               <br> &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; apiConsumer.BaseUri = new Uri(url);
               <br> &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; apiConsumer.ContentType = ApiConsumer.ApplicationJson;

               <br> &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; &nbsp; }
