         function member_interaction(op, id, id2) {

            var $url = window.top.location,
                $data, // op: operation; cid: comment id; mid: comment's owner member id; ao: ajax operation
                $dataType = 'html',
                bool = true,
                cmt; // comments

            switch (op) {

                     $data = {op: op, cid: id, mid: id2, ao: 1};
                  } else {

                  }

                  cmt = $.trim($('#inp0').val());

                  if (cmt === '') {
                     $('#inp0').focus();
                     error_msg("main", "D'oh! You have to enter something to post a comment.");
                     return false;
                  } else {
                     $data = {op: 'ac', cmt: cmt, ao: 1};
                  }

                  break;

               case 'ac': // reply comment
                  cmt = $.trim($('#inp' + id).val());

                  if (cmt === '' || cmt === "Write a comment...") {
                     $('#inp' + id).focus();

                     $data = {op: op, cid: id, cmt: cmt, mid: id2, ao: 1};

                  $data = {op: op, ao: 1};

                     window.location = 'http://gallery.mobile9.com/upload/?fid=37631&op3=dlt';

                     return false;

       // create new collection
        function create_collection() {

            $('.red').remove(); // remove existing error msg
            $('textarea').removeClass('redBox'); //remove other element existing red color border for textarea
            $('#collection_field').removeClass('redBox'); // remove red border

            var name = $('#collection_field').val(); // get the value of collection name

            if (name === '') {
               $('#collection_field').addClass('redBox').focus(); // add red border, focus input field
               $("<div class=\"red\">D'oh! You have to enter something to create collection name.</div>").insertAfter('#create_form'); // add error msg
            } else {
               $.ajax({
                  type: 'POST',
                  url: window.top.location,
                  data: {op: 'aco', ao: 1, conm: name},
                  dataType: 'json',
                  beforeSend: ShowLoading,
                  success: function (data) {
                        $('#theLoading').dialog('close');

                        if (data.status === 1) { // success

                           if (data.del !== 0) {
                              $radioInput = $('#collection_list :radio');
                              $.each($radioInput, function(idx) {
                                    if ($(this).val() === data.del) {
                                       $(this).parent().remove();
                                    }
                              });
                           }

                           $('.collection ul').append(
                                                 '<li>\n\
                                                   <input type="radio" id="' + data.un_conm + '" name="coid" value="' + data.coid + '" onclick="change_collection(this.value);" />\n\
                                                   <label for="' + data.un_conm + '">' + data.conm + '</label> \n\
                                                 </li>');
                           $radioInput = $('#collection_list :radio'); // radio input
                           $radioInput.removeAttr('checked'); // uncheck all radio input
                           $.each($radioInput, function(idx) {
                              if ($(this).val() === data.coid) {
                                    $(this).attr('checked', 'checked');
                              }
                           });
                           $('#collection_field').focus().val(''); // focus and clear input field
                        } else if (data.status === 0) { // fail
                           $('#collection_field').addClass('redBox').focus().val(''); // add red border, focus and clear input field
                           $("<div class=\"red\">" + data.sysresp + "</div>").insertAfter('#create_form'); // add error msg
                        }
                  }
               });
            }

            return false;
        } // create_collection()

        // member report: file or comment
        function member_report(type, id, report_type) {

            //id: comment id / file id
            var re_url = 'http://gallery.mobile9.com/rp/?';

            $('.sysresp').remove(); // remove existing display $sysresp

            $('#ty').val('gallery');

            if (type === 'reportFile') {

               var obj = "<span class=\"gallery_icons valign_center reported\"></span><span class=\"valign_center sub_grey_text ml3 uppercase\">REPORTED</span>", //change inner text to "reported";
                   rty = $( "input:checked" ).val(), //input report radio button
                  link = $('#report_url').val(),//duplicate url
              trim_url = $.trim(link);

              if ($("input#Duplicate").is(':checked')){

                    if (trim_url ==""){
                        error_msg("report_url", "Please enter your original file link.")
                        return false;
                    } else{
                        if (trim_url == 'http://gallery.mobile9.com/f/37631/'){
                           error_msg("report_url", "Oops, this is the same file link. Please enter the original file link.");
                           return false;
                        } else{
                            url_validate = /^(?:http?:\/\/)?gallery\.mobile9\.com/;
                            if(!url_validate.test(trim_url)){
                                error_msg("report_url", "Invalid URL. E.g. http://gallery.mobile9.com/f/3440855/");
                                return false;
                            }
                       }
                    }//trim_url
              }


               $('#rty').val(rty); //input hidden
               $('#ofl').val(link); //input hidden

               $('#report-btn').html(obj).addClass('been_reported');
               //$('#report_type_' + id).hide();   //hide dropdown div
               recalc_position('.user_row1', 'line' + line_no++);
               $('#cat').val("softwares");

               url = re_url + 'fid=' + id + '&';

            } else { // report comment
               $('#rty').val(report_type);
               $('#' + type + id).unbind('click');
               $('#' + type + id).html('Reported. Thanks!'); //change inner text to "reported";
               $('#' + type + id).removeClass('report_content').addClass('grey_text');  //reset class name for report comment

               $('#cat').val('comments');

               url = re_url + 'cid=' + id + '&';

            }

            update_tpl2('interaction', type);

            return false;

        } // member_report()
