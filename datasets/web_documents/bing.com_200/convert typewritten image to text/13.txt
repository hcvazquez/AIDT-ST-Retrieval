             <div id="ctl00_description" class="summary">
              Convert Text to Image
             </div>
             <span id="ctl00_thumbnailUrl" class="date" content="https://www.codeproject.com/script/Articles/Images/tip100x80.png"></span> 
            </div> 
            <div id="contentdiv" class="text">
              We can easily convert text to image in C# by using 
             <code>System.Drawing</code> namespace. Here these tips will guide you how to create bitmap image from text. It’s using 
             <code>Graphics</code> property of 
             <code>System.Drawing</code> namespace to convert text to bitmap image and place the image in a picture box. The 
             <code>Graphics</code> class provides methods for drawing to the display device.
             <br> In this article, you will see a method 
             <code>Convert_Text_to_Image(string txt, string fontname, int fontsize)</code> is responsible to return Bitmap Image on the basis of 
             <b>Text</b>, 
             <b>fontname</b> and 
             <b>fontsize</b>. 
             <br> Take a look at how to use the mentioned method for converting Text to Image: 
             <br> 
             <br> 
             <pre><span class="code-keyword">using</span> System;
<span class="code-keyword">using</span> System.Collections.Generic;
<span class="code-keyword">using</span> System.ComponentModel;
<span class="code-keyword">using</span> System.Data;
<span class="code-keyword">using</span> System.Drawing;
<span class="code-keyword">using</span> System.Text;
<span class="code-keyword">using</span> System.Windows.Forms;
&nbsp;
<span class="code-keyword">namespace</span> Text_to_Image
{
    <span class="code-keyword">public</span> <span class="code-keyword">partial</span> <span class="code-keyword">class</span> Form1 : Form
    {
        <span class="code-keyword">public</span> Form1()
        {
            InitializeComponent();
        }
&nbsp;
        <span class="code-keyword">public</span> <span class="code-keyword">static</span> Bitmap Convert_Text_to_Image(<span class="code-keyword">string</span> txt, <span class="code-keyword">string</span> fontname, <span class="code-keyword">int</span> fontsize)
        {
            <span class="code-comment">//</span><span class="code-comment">creating bitmap image</span>
            Bitmap bmp = <span class="code-keyword">new</span> Bitmap(<span class="code-digit">1</span>, <span class="code-digit">1</span>);
           
            <span class="code-comment">//</span><span class="code-comment">FromImage method creates a new Graphics from the specified Image.</span>
            Graphics graphics = Graphics.FromImage(bmp);
            <span class="code-comment">//</span><span class="code-comment"> Create the Font object for the image text drawing.</span>
            Font font = <span class="code-keyword">new</span> Font(fontname, fontsize);
           <span class="code-comment">//</span><span class="code-comment"> Instantiating object of Bitmap image again with the correct size for the text and font.</span>
            SizeF stringSize = graphics.MeasureString(txt, font);
            bmp = <span class="code-keyword">new</span> Bitmap(bmp,(<span class="code-keyword">int</span>)stringSize.Width,(<span class="code-keyword">int</span>)stringSize.Height); 
            graphics = Graphics.FromImage(bmp);
&nbsp;
           <span class="code-comment">/*</span><span class="code-comment"> It can also be a way
          bmp = new Bitmap(bmp, new Size((int)graphics.MeasureString(txt, font).Width, (int)graphics.MeasureString(txt, font).Height));*/</span>
&nbsp;
            <span class="code-comment">//</span><span class="code-comment">Draw Specified text with specified format </span>
            graphics.DrawString(txt, font, Brushes.Red, <span class="code-digit">0</span>, <span class="code-digit">0</span>);
            font.Dispose();            
            graphics.Flush();
            graphics.Dispose();
            <span class="code-keyword">return</span> bmp;     <span class="code-comment">//</span><span class="code-comment">return Bitmap Image </span>
         }
&nbsp;
        <span class="code-keyword">private</span> <span class="code-keyword">void</span> btnconvert_Click(<span class="code-keyword">object</span> sender, EventArgs e)
        {
           picbox.Image = Convert_Text_to_Image(txtvalue.Text, <span class="code-string">"</span><span class="code-string">Bookman Old Style"</span>, <span class="code-digit">20</span>); <span class="code-comment">//</span><span class="code-comment"> Passing appropriate value to Convert_Text_to_Image method </span>
           picbox.SizeMode = PictureBoxSizeMode.StretchImage;
        }      
    }
}</pre>
             <br> Now write something in 
             <code>Textbox</code> and then press convert button to see the desired output.
             <br> Go 
             <a href="https://docs.google.com/leaf?id=0B3AkJApDa7oCMzllYTk1YmItY2E4OC00ZjQ1LWFjNzEtZmE1MTQ2ZTc4N2Y0&amp;hl=en&amp;authkey=COjNipgJ">there</a>[
             <a href="https://docs.google.com/leaf?id=0B3AkJApDa7oCMzllYTk1YmItY2E4OC00ZjQ1LWFjNzEtZmE1MTQ2ZTc4N2Y0&amp;hl=en&amp;authkey=COjNipgJ" target="_blank" title="New Window">^</a>] to download source code.
             <br> 
             <br> 
             <br> 
             <b>[Edited]</b> Another approach 
             <br> 
             <pre><span class="code-keyword">public</span> Bitmap ConvertTextToImage(<span class="code-keyword">string</span> txt, <span class="code-keyword">string</span> fontname, <span class="code-keyword">int</span> fontsize, Color bgcolor, Color fcolor, <span class="code-keyword">int</span> width, <span class="code-keyword">int</span> Height)
        {
            Bitmap bmp = <span class="code-keyword">new</span> Bitmap(width, Height);
            <span class="code-keyword">using</span> (Graphics graphics = Graphics.FromImage(bmp))
            {
&nbsp;

                              <td colspan="2">private Image DrawText(String text, Font font, Color textColor, Color backColor)<br> {<br> //first, create a dummy bitmap just to get a graphics object<br> Image img = new Bitmap(1, 1);<br> Graphics drawing = Graphics.FromImage(img);<br> <br> //measure the string to see how big the image needs to be<br> SizeF textSize = drawing.MeasureString(text, font);<br> <br> //free up the dummy image and old graphics object<br> img.Dispose();<br> drawing.Dispose();<br> <br> //create a new image of the right size<br> img = new Bitmap((int) textSize.Width, (int)textSize.Height);<br> <br> drawing = Graphics.FromImage(img);<br> <br> //paint the background<br> drawing.Clear(backColor);<br> <br> //create a brush for the text<br> Brush textBrush = new SolidBrush(textColor);<br> <br> drawing.DrawString(text, font, textBrush, 0, 0);<br> <br> drawing.Save();<br> <br> textBrush.Dispose();<br> drawing.Dispose();<br> <br> return img;<br> <br> }<br>
