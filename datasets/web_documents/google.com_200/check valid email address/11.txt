        <h1 id="accounts.resendVerificationCodeJS-Description">Description</h1>
        <p>This method is used to resend a validation email to unverified addresses associated with the account.&nbsp;The email format is according to the templates defined in the policy.&nbsp;For more information on the email format, please refer to&nbsp;<a href="/display/GD/accounts.setPolicies+REST">account.setPolicies</a>&nbsp;or to the&nbsp;<a href="/display/GD/Policies#Policies-EmailTemplates">Email Templates</a>&nbsp;section of the&nbsp;User Management <a href="/display/GD/Policies">Policies</a>&nbsp;guide.</p>
        <p><span></span></p>
        <div class="panel conf-macro output-block conf-macro output-block" data-hasbody="false" data-macro-name="include" style="background-color: #F4F7FC;border-color: #81C0FC;border-style: solid;border-width: 1px;">
         <div class="panelContent" style="background-color: #F4F7FC;"> 
          <p><strong>Note:</strong>&nbsp;If you plan on integrating Gigya's Accounts API, we highly recommend reading the&nbsp;<a href="/display/GD/Registration-as-a-Service">Registration-as-a-Service</a>&nbsp;Guide. <a href="/display/GD/Registration-as-a-Service">Registration-as-a-Service</a>&nbsp;(RaaS) is a premium platform that requires separate activation. If <strong>RaaS</strong> is not part of your site package, please contact Gigya by filling in a&nbsp;<a class="external-link" href="https://console.gigya.com/Site/partners/Support.aspx" rel="nofollow" title="https://console.gigya.com/Site/partners/Support.aspx">support form</a>&nbsp;through the Console. You can access the support page by clicking <strong><a class="external-link" href="https://console.gigya.com/site/partners/SupportPortal.aspx" rel="nofollow">Support</a></strong> on the upper menu after logging into the Gigya Console.</p>

        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1);
        if (c.indexOf(name) == 0) return c.substring(name.length,c.length);
    }
    return "";
}
