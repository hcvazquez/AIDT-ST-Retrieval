              <p>Don't forget to handle errors! The following code is based on Augusto Roman's answer.</p> 
              <pre><code>var http = require('http');
var fs = require('fs');

var download = function(url, dest, cb) {
  var file = fs.createWriteStream(dest);
  var request = http.get(url, function(response) {
    response.pipe(file);
    file.on('finish', function() {
      file.close(cb);  // close() is async, call cb after close completes.
    });
  }).on('error', function(err) { // Handle errors
    fs.unlink(dest); // Delete the file async. (But we don't check the result)
    if (cb) cb(err.message);
  });
};
</code></pre>

              <p>As Brandon Tilley said, but with the appropriate control flow:</p> 
              <pre><code>var http = require('http');
var fs = require('fs');

var download = function(url, dest, cb) {
  var file = fs.createWriteStream(dest);
  var request = http.get(url, function(response) {
    response.pipe(file);
    file.on('finish', function() {
      file.close(cb);
    });
  });
}
</code></pre> 
              <p>Without waiting for the <code>finish</code> event, naive scripts may end up with an incomplete file.</p> 
              <p><strong>Edit:</strong> Thanks to @Augusto Roman for pointing out that <code>cb</code> should be passed to <code>file.close</code>, not called explicitly.</p>

             <div class="post-text" itemprop="text"> 
              <p>Speaking of handling errors, it's even better listening to request errors too. I'd even validate by checking response code. Here it's considered success only for 200 response code, but other codes might be good. </p> 
              <pre><code>var fs = require('fs');
var http = require('http');

var download = function(url, dest, cb) {
    var file = fs.createWriteStream(dest);
    var request = http.get(url, function(response) {

        // check if response is success
        if (response.statusCode !== 200) {
            return cb('Response status was ' + response.statusCode);
        }

        response.pipe(file);

        file.on('finish', function() {
            file.close(cb);  // close() is async, call cb after close completes.
        });
    });

    // check for request error too
    request.on('error', function (err) {
        fs.unlink(dest);
        return cb(err.message);
    });

    file.on('error', function(err) { // Handle errors
        fs.unlink(dest); // Delete the file async. (But we don't check the result) 
        return cb(err.message);
    });
};
</code></pre> 
              <p>Despite the relative simplicity of this code, I would advise to use the <a href="https://github.com/request/request" rel="noreferrer">request module</a> as it handles much more protocols (hello HTTPS!) which aren't natively supported by <code>http</code>.</p> 
              <p>That would be done like so:</p> 
              <pre><code>var fs = require('fs');
var request = require('request');

var download = function(url, dest, cb) {
    var file = fs.createWriteStream(dest);
    var sendReq = request.get(url);

    // verify response code
    sendReq.on('response', function(response) {
        if (response.statusCode !== 200) {
            return cb('Response status was ' + response.statusCode);
        }
    });

    // check for request errors
    sendReq.on('error', function (err) {
        fs.unlink(dest);
        return cb(err.message);
    });

    sendReq.pipe(file);

    file.on('finish', function() {
        file.close(cb);  // close() is async, call cb after close completes.
    });

    file.on('error', function(err) { // Handle errors
        fs.unlink(dest); // Delete the file async. (But we don't check the result)
        return cb(err.message);
    });
};
</code></pre>

            <td class="answercell"> 
             <div class="post-text" itemprop="text"> 
              <p>gfxmonk's answer has a very tight data race between the callback and the file.close() completing. file.close() actually takes a callback that is called when the close has completed. Otherwise, immediate uses of the of the file may fail (very rarely!).</p> 
              <p>A complete solution is:</p> 
              <pre><code>var http = require('http');
var fs = require('fs');

var download = function(url, dest, cb) {
  var file = fs.createWriteStream(dest);
  var request = http.get(url, function(response) {
    response.pipe(file);
    file.on('finish', function() {
      file.close(cb);  // close() is async, call cb after close completes.
    });
  });
}
</code></pre> 
              <p>Without waiting for the finish event, naive scripts may end up with an incomplete file. Without scheduling the cb callback via close, you may get a race between accessing the file and the file actually being ready.</p>

              <h1>Solution with timeout, prevent memory leak :</h1> 
              <p>The following code is based on Brandon Tilley's answer :</p> 
              <pre><code>var http = require('http'),
    fs = require('fs');

var request = http.get("yourfile.html", function(response) {
    if (response.statusCode === 200) {
        var file = fs.createWriteStream("copy.html");
        response.pipe(file);
    }
    // Add timeout.
    request.setTimeout(12000, function () {
        request.abort();
    });
});
</code></pre> 
              <p>Don't make file when you get an error, and prefere to use timeout to close your request after X secondes.</p>

              <p>for those who came in search of es6-style promise based way, I guess it would be something like: </p> 
              <pre><code>var http = require('http');
var fs = require('fs');

function pDownload(url, dest){
  var file = fs.createWriteStream(dest);
  return new Promise((resolve, reject) =&gt; {
    var responseSent = false; // flag to make sure that response is sent only once.
    http.get(url, response =&gt; {
      response.pipe(file);
      file.on('finish', () =&gt;{
        file.close(() =&gt; {
          if(responseSent)  return;
          responseSent = true;
          resolve();
        });
      });
    }).on('error', err =&gt; {
        if(responseSent)  return;
        responseSent = true;
        reject(err);
    });
  });
}

//example
pDownload(url, fileLocation)
  .then( ()=&gt; console.log('downloaded file no issues...'))
  .catch( e =&gt; console.error('error while downloading', e));
</code></pre>

              <p>Download using promise, which resolve a readable stream. put extra logic to handle the redirect.</p> 
              <pre><code>var http = require('http');
var promise = require('bluebird');
var url = require('url');
var fs = require('fs');
var assert = require('assert');

function download(option) {
    assert(option);
    if (typeof option == 'string') {
        option = url.parse(option);
    }

    return new promise(function(resolve, reject) {
        var req = http.request(option, function(res) {
            if (res.statusCode == 200) {
                resolve(res);
            } else {
                if (res.statusCode === 301 &amp;&amp; res.headers.location) {
                    resolve(download(res.headers.location));
                } else {
                    reject(res.statusCode);
                }
            }
        })
        .on('error', function(e) {
            reject(e);
        })
        .end();
    });
}

download('http://localhost:8080/redirect')
.then(function(stream) {
    try {

        var writeStream = fs.createWriteStream('holyhigh.jpg');
        stream.pipe(writeStream);

    } catch(e) {
        console.error(e);
    }
});
</code></pre>

            <td class="answercell"> 
             <div class="post-text" itemprop="text"> 
              <p>Maybe node.js has changed, but it seems there are some problems with the other solutions (using node v8.1.2):</p> 
              <ol> 
               <li>You don't need to call <code>file.close()</code> in the <code>finish</code> event. Per default the <code>fs.createWriteStream</code> is set to autoClose: <a href="https://nodejs.org/api/fs.html#fs_fs_createwritestream_path_options" rel="nofollow noreferrer">https://nodejs.org/api/fs.html#fs_fs_createwritestream_path_options</a></li> 
               <li><code>file.close()</code> should be called on error. Maybe this is not needed when the file is deleted (<code>unlink()</code>), but normally it is: <a href="https://nodejs.org/api/stream.html#stream_readable_pipe_destination_options" rel="nofollow noreferrer">https://nodejs.org/api/stream.html#stream_readable_pipe_destination_options</a></li> 
               <li>Temp file is not deleted on <code>statusCode !== 200</code></li> 
               <li><code>fs.unlink()</code> without a callback is deprecated (outputs warning)</li> 
               <li>If <code>dest</code> file exists; it is overridden</li> 
              </ol> 
              <p>Below is a modified solution (using ES6 and promises) which handles these problems.</p> 
              <pre><code>const http = require("http");
const fs = require("fs");

function download(url, dest) {
    return new Promise((resolve, reject) =&gt; {
        const file = fs.createWriteStream(dest, { flags: "wx" });

        const request = http.get(url, response =&gt; {
            if (response.statusCode === 200) {
                response.pipe(file);
            } else {
                file.close();
                fs.unlink(dest, () =&gt; {}); // Delete temp file
                reject(`Server responded with ${response.statusCode}: ${response.statusMessage}`);
            }
        });

        request.on("error", err =&gt; {
            file.close();
            fs.unlink(dest, () =&gt; {}); // Delete temp file
            reject(err.message);
        });

        file.on("finish", () =&gt; {
            resolve();
        });

        file.on("error", err =&gt; {
            file.close();

            if (err.code === "EEXIST") {
                reject("File already exists");
            } else {
                fs.unlink(dest, () =&gt; {}); // Delete temp file
                reject(err.message);
            }
        });
    });
}
</code></pre>
